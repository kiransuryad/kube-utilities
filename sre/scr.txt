#!/bin/bash

set -e

REGION=${AWS_REGION:-eu-west-1}
OUT_DIR="evidence-before"
mkdir -p "$OUT_DIR"

echo "Starting BEFORE evidence capture..."
echo "Region: $REGION"
echo

pause_and_capture() {
  NAME=$1
  echo
  echo "======================================"
  echo "Captured: $NAME"
  echo "Time: $(date)"
  echo "======================================"
  sleep 5
  screencapture -x "$OUT_DIR/${NAME}.png"
}

# -----------------------------
# RDS Instances
# -----------------------------
echo "RDS Instances:"
aws rds describe-db-instances \
  --query 'DBInstances[].DBInstanceIdentifier' \
  --output table | tee "$OUT_DIR/rds-instances.txt"

pause_and_capture "01-rds-instances-before"

# -----------------------------
# RDS Snapshots
# -----------------------------
echo "RDS Snapshots:"
aws rds describe-db-snapshots \
  --query 'DBSnapshots[].DBSnapshotIdentifier' \
  --output table | tee "$OUT_DIR/rds-snapshots.txt"

pause_and_capture "02-rds-snapshots-before"

# -----------------------------
# S3 Buckets
# -----------------------------
echo "S3 Buckets:"
aws s3 ls | tee "$OUT_DIR/s3-buckets.txt"

pause_and_capture "03-s3-buckets-before"

# -----------------------------
# DynamoDB Tables
# -----------------------------
echo "DynamoDB Tables:"
aws dynamodb list-tables \
  --query 'TableNames' \
  --output table | tee "$OUT_DIR/dynamodb-tables.txt"

pause_and_capture "04-dynamodb-before"

# -----------------------------
# EBS Snapshots
# -----------------------------
echo "EBS Snapshots:"
aws ec2 describe-snapshots --owner-ids self \
  --query 'Snapshots[].SnapshotId' \
  --output table | tee "$OUT_DIR/ebs-snapshots.txt"

pause_and_capture "05-ebs-snapshots-before"

# -----------------------------
# EFS File Systems
# -----------------------------
echo "EFS File Systems:"
aws efs describe-file-systems \
  --query 'FileSystems[].FileSystemId' \
  --output table | tee "$OUT_DIR/efs-before.txt"

pause_and_capture "06-efs-before"

# -----------------------------
# Backup Vaults
# -----------------------------
echo "Backup Vaults:"
aws backup list-backup-vaults \
  --query 'BackupVaultList[].BackupVaultName' \
  --output table | tee "$OUT_DIR/backup-vaults.txt"

pause_and_capture "07-backup-vaults-before"

echo
echo "✅ BEFORE evidence capture completed"
echo "Saved in: $OUT_DIR"

#!/usr/bin/env python3

import json
import sys
from pathlib import Path
from datetime import datetime

import pandas as pd


def load_json(file_path):
    try:
        with open(file_path, "r") as f:
            return json.load(f)
    except Exception:
        return {}


def rows_from_rds_instances(data, phase, file_name):
    rows = []
    for item in data.get("DBInstances", []):
        rows.append({
            "Service": "RDS",
            "Resource Type": "DB Instance",
            "Resource Name": item.get("DBInstanceIdentifier"),
            "Region/Scope": item.get("AvailabilityZone", ""),
            "Status": item.get("DBInstanceStatus"),
            "Engine": item.get("Engine"),
            "Phase": phase,
            "Evidence File": file_name,
        })
    return rows

PARSERS = {
    "rds-db-instances.json": rows_from_rds_instances,
    "rds-db-snapshots.json": rows_from_rds_snapshots,
    "s3-buckets.json": rows_from_s3_buckets,
    "dynamodb-tables.json": rows_from_dynamodb_tables,
    "dynamodb-backups.json": rows_from_dynamodb_backups,
    "ebs-snapshots.json": rows_from_ebs_snapshots,
    "efs-filesystems.json": rows_from_efs,
    "backup-vaults.json": rows_from_backup_vaults,
    "ec2-instances.json": rows_from_ec2_instances,
}


def detect_phase(path):
    name = path.name.lower()
    full = str(path).lower()

    if "before" in name or "before" in full:
        return "Before"
    if "after" in name or "after" in full:
        return "After"

    return "Unknown"


def collect_rows(input_dir):
    input_path = Path(input_dir)
    all_rows = []

    for json_file in input_path.rglob("*.json"):
        parser = PARSERS.get(json_file.name)
        if not parser:
            continue

        data = load_json(json_file)
        phase = detect_phase(json_file)
        rows = parser(data, phase, json_file.name)
        all_rows.extend(rows)

    return all_rows


def create_summary(df):
    if df.empty:
        return pd.DataFrame(columns=["Service", "Resource Type", "Before Count", "After Count", "Status"])

    grouped = (
        df.groupby(["Service", "Resource Type", "Phase"])
        .size()
        .reset_index(name="Count")
    )

    pivot = grouped.pivot_table(
        index=["Service", "Resource Type"],
        columns="Phase",
        values="Count",
        fill_value=0
    ).reset_index()

    if "Before" not in pivot.columns:
        pivot["Before"] = 0
    if "After" not in pivot.columns:
        pivot["After"] = 0

    pivot = pivot.rename(columns={
        "Before": "Before Count",
        "After": "After Count"
    })

    pivot["Status"] = pivot.apply(
        lambda r: "Cleared" if r["Before Count"] > 0 and r["After Count"] == 0
        else "Still Present" if r["After Count"] > 0
        else "No Resources Found",
        axis=1
    )

    return pivot[["Service", "Resource Type", "Before Count", "After Count", "Status"]]


def autosize_excel(writer, sheet_name, df):
    worksheet = writer.sheets[sheet_name]

    for idx, col in enumerate(df.columns):
        max_len = max(
            [len(str(col))] + [len(str(x)) for x in df[col].astype(str).values]
        )
        worksheet.set_column(idx, idx, min(max_len + 2, 50))


def main():
    if len(sys.argv) < 2:
        print("Usage: python aws_evidence_to_excel.py <evidence-folder>")
        sys.exit(1)

    input_dir = sys.argv[1]
    timestamp = datetime.now().strftime("%Y%m%d-%H%M%S")
    output_file = f"aws-decommission-evidence-{timestamp}.xlsx"

    rows = collect_rows(input_dir)
    inventory_df = pd.DataFrame(rows)

    if inventory_df.empty:
        print("No supported AWS evidence JSON files found.")
        sys.exit(1)

    summary_df = create_summary(inventory_df)

    with pd.ExcelWriter(output_file, engine="xlsxwriter") as writer:
        summary_df.to_excel(writer, sheet_name="Summary", index=False)
        inventory_df.to_excel(writer, sheet_name="Resource Inventory", index=False)

        before_df = inventory_df[inventory_df["Phase"] == "Before"]
        after_df = inventory_df[inventory_df["Phase"] == "After"]

        before_df.to_excel(writer, sheet_name="Before Evidence", index=False)
        after_df.to_excel(writer, sheet_name="After Evidence", index=False)

        autosize_excel(writer, "Summary", summary_df)
        autosize_excel(writer, "Resource Inventory", inventory_df)
        autosize_excel(writer, "Before Evidence", before_df)
        autosize_excel(writer, "After Evidence", after_df)

    print(f"Excel evidence workbook created: {output_file}")


if __name__ == "__main__":
    main()

#!/usr/bin/env bash
set -euo pipefail

PHASE="${1:-before}"   # before or after
ACCOUNT_ID="$(aws sts get-caller-identity --query Account --output text)"
REGION="${AWS_REGION:-$(aws configure get region)}"
TIMESTAMP="$(date +%Y%m%d-%H%M%S)"

OUT_DIR="evidence-${ACCOUNT_ID}-${REGION}-${PHASE}-${TIMESTAMP}"
mkdir -p "$OUT_DIR"

echo "Capturing AWS decommission evidence"
echo "Account: $ACCOUNT_ID"
echo "Region:  $REGION"
echo "Phase:   $PHASE"
echo "Output:  $OUT_DIR"

run_cmd () {
  NAME="$1"
  shift

  echo "Running: $NAME"
  {
    echo "Command: $*"
    echo "Captured at: $(date -Iseconds)"
    echo
    "$@"
  } 2>&1 | tee "$OUT_DIR/${NAME}.txt"
}

run_json () {
  NAME="$1"
  shift

  echo "Running: $NAME"
  "$@" 2>&1 | tee "$OUT_DIR/${NAME}.json" >/dev/null || true
}

# Identity
run_json "identity" aws sts get-caller-identity

# RDS
run_json "rds-db-instances" aws rds describe-db-instances
run_json "rds-db-snapshots" aws rds describe-db-snapshots --snapshot-type manual
run_json "rds-clusters" aws rds describe-db-clusters
run_json "rds-cluster-snapshots" aws rds describe-db-cluster-snapshots --snapshot-type manual

# S3
run_json "s3-buckets" aws s3api list-buckets

# DynamoDB
run_json "dynamodb-tables" aws dynamodb list-tables
run_json "dynamodb-backups" aws dynamodb list-backups

# EBS snapshots
run_json "ebs-snapshots" aws ec2 describe-snapshots --owner-ids self

# EFS
run_json "efs-filesystems" aws efs describe-file-systems

# AWS Backup
run_json "backup-vaults" aws backup list-backup-vaults

# Extra useful cost/resource indicators
run_json "ec2-instances" aws ec2 describe-instances
run_json "ebs-volumes" aws ec2 describe-volumes
run_json "eks-clusters" aws eks list-clusters
run_json "load-balancers-v2" aws elbv2 describe-load-balancers
run_json "nat-gateways" aws ec2 describe-nat-gateways
run_json "elastic-ips" aws ec2 describe-addresses

# Human-readable summary
{
  echo "# AWS Decommission Evidence Summary"
  echo
  echo "Account: $ACCOUNT_ID"
  echo "Region: $REGION"
  echo "Phase: $PHASE"
  echo "Captured at: $(date -Iseconds)"
  echo
  echo "## Key checks"
  echo
  echo "RDS instances:"
  jq -r '.DBInstances[]?.DBInstanceIdentifier // empty' "$OUT_DIR/rds-db-instances.json" || true
  echo
  echo "RDS manual snapshots:"
  jq -r '.DBSnapshots[]?.DBSnapshotIdentifier // empty' "$OUT_DIR/rds-db-snapshots.json" || true
  echo
  echo "S3 buckets:"
  jq -r '.Buckets[]?.Name // empty' "$OUT_DIR/s3-buckets.json" || true
  echo
  echo "DynamoDB tables:"
  jq -r '.TableNames[]? // empty' "$OUT_DIR/dynamodb-tables.json" || true
  echo
  echo "EBS snapshots:"
  jq -r '.Snapshots[]?.SnapshotId // empty' "$OUT_DIR/ebs-snapshots.json" || true
  echo
  echo "EFS file systems:"
  jq -r '.FileSystems[]?.FileSystemId // empty' "$OUT_DIR/efs-filesystems.json" || true
  echo
  echo "Backup vaults:"
  jq -r '.BackupVaultList[]?.BackupVaultName // empty' "$OUT_DIR/backup-vaults.json" || true
} | tee "$OUT_DIR/SUMMARY.md"

tar -czf "${OUT_DIR}.tar.gz" "$OUT_DIR"

echo
echo "Evidence captured successfully:"
echo "$OUT_DIR"
echo "${OUT_DIR}.tar.gz"
