#!/bin/bash

set -u

REGION="us-east-1"
MODE="${1:-dry-run}"

log() {
  echo "[INFO] $*"
}

run_cmd() {
  echo
  echo "COMMAND: $*"
  echo

  if [ "$MODE" = "--execute" ]; then
    "$@"
  else
    echo "DRY RUN ONLY"
  fi
}

confirm_mode() {
  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY RUN mode"
    log "Execute with:"
    echo "./delete-vpc-links-and-endpoint-services.sh --execute"
    exit 0
  fi

  echo
  read -p "Type DELETE-VPC-LINKS to continue: " CONFIRM

  if [ "$CONFIRM" != "DELETE-VPC-LINKS" ]; then
    echo "Cancelled."
    exit 1
  fi
}

delete_vpc_links() {

  log "Discovering API Gateway VPC Links"

  VPC_LINK_IDS=$(aws apigateway get-vpc-links \
    --region "$REGION" \
    --query 'items[].id' \
    --output text)

  if [ -z "$VPC_LINK_IDS" ]; then
    log "No VPC Links found."
    return
  fi

  for ID in $VPC_LINK_IDS; do

    NAME=$(aws apigateway get-vpc-link \
      --region "$REGION" \
      --vpc-link-id "$ID" \
      --query 'name' \
      --output text)

    STATUS=$(aws apigateway get-vpc-link \
      --region "$REGION" \
      --vpc-link-id "$ID" \
      --query 'status' \
      --output text)

    log "Deleting VPC Link:"
    echo "ID: $ID"
    echo "Name: $NAME"
    echo "Status: $STATUS"

    run_cmd aws apigateway delete-vpc-link \
      --region "$REGION" \
      --vpc-link-id "$ID"
  done
}

delete_endpoint_services() {

  log "Discovering VPC Endpoint Service Configurations"

  SERVICE_IDS=$(aws ec2 describe-vpc-endpoint-service-configurations \
    --region "$REGION" \
    --query 'ServiceConfigurations[].ServiceId' \
    --output text)

  if [ -z "$SERVICE_IDS" ]; then
    log "No VPC Endpoint Services found."
    return
  fi

  for ID in $SERVICE_IDS; do

    NAME=$(aws ec2 describe-vpc-endpoint-service-configurations \
      --region "$REGION" \
      --service-ids "$ID" \
      --query 'ServiceConfigurations[0].ServiceName' \
      --output text)

    STATE=$(aws ec2 describe-vpc-endpoint-service-configurations \
      --region "$REGION" \
      --service-ids "$ID" \
      --query 'ServiceConfigurations[0].ServiceState' \
      --output text)

    log "Deleting Endpoint Service:"
    echo "ID: $ID"
    echo "Name: $NAME"
    echo "State: $STATE"

    run_cmd aws ec2 delete-vpc-endpoint-service-configurations \
      --region "$REGION" \
      --service-ids "$ID"
  done
}

main() {

  log "Starting VPC Link + Endpoint Service cleanup"
  log "Region: $REGION"

  confirm_mode

  delete_vpc_links

  echo
  log "Waiting 60 seconds before deleting endpoint services"
  sleep 60

  delete_endpoint_services

  echo
  log "Cleanup completed."
  log "Now wait 5-10 minutes and rerun:"
  echo "./delete-leftover-elbv2-resources.sh --execute"
}

#!/bin/bash

set -u

REGION="us-east-1"
MODE="${1:-dry-run}"
OUT_DIR="elbv2-cleanup-logs"
RUN_DATE=$(date '+%Y-%m-%d')
LOG_FILE="$OUT_DIR/delete-leftover-elbv2-${RUN_DATE}.log"

mkdir -p "$OUT_DIR"

log() {
  echo "[$RUN_DATE] $*" | tee -a "$LOG_FILE"
}

run_cmd() {
  log "COMMAND: $*"

  if [ "$MODE" = "--execute" ]; then
    "$@" 2>&1 | tee -a "$LOG_FILE" || true
  else
    log "DRY-RUN ONLY"
  fi

  echo | tee -a "$LOG_FILE"
}

confirm_mode() {
  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY-RUN mode."
    log "To execute: ./delete-leftover-elbv2-resources.sh --execute"
    echo
  else
    log "RUNNING ELBV2 CLEANUP IN EXECUTE MODE."
    read -p "Type DELETE-ELBV2 to continue: " CONFIRM
    if [ "$CONFIRM" != "DELETE-ELBV2" ]; then
      log "Cancelled."
      exit 1
    fi
  fi
}

delete_listeners_for_lb() {
  LB_ARN="$1"

  LISTENERS=$(aws elbv2 describe-listeners \
    --region "$REGION" \
    --load-balancer-arn "$LB_ARN" \
    --query 'Listeners[].ListenerArn' \
    --output text 2>/dev/null)

  if [ -z "$LISTENERS" ]; then
    log "No listeners found for LB: $LB_ARN"
    return
  fi

  for LISTENER in $LISTENERS; do
    log "Deleting listener: $LISTENER"
    run_cmd aws elbv2 delete-listener \
      --region "$REGION" \
      --listener-arn "$LISTENER"
  done
}

delete_leftover_load_balancers() {
  log "Checking leftover Kubernetes/EKS load balancers"

  LBS=$(aws elbv2 describe-load-balancers \
    --region "$REGION" \
    --query "LoadBalancers[?starts_with(LoadBalancerName, 'k8s-') || starts_with(LoadBalancerName, 'internal-k8s-') || contains(LoadBalancerName, 'backend') || contains(LoadBalancerName, 'ingress') || contains(LoadBalancerName, 'internal')].LoadBalancerArn" \
    --output text 2>/dev/null)

  if [ -z "$LBS" ]; then
    log "No leftover Kubernetes/EKS-looking load balancers found."
    return
  fi

  for LB in $LBS; do
    NAME=$(aws elbv2 describe-load-balancers \
      --region "$REGION" \
      --load-balancer-arns "$LB" \
      --query 'LoadBalancers[0].LoadBalancerName' \
      --output text 2>/dev/null)

    TYPE=$(aws elbv2 describe-load-balancers \
      --region "$REGION" \
      --load-balancer-arns "$LB" \
      --query 'LoadBalancers[0].Type' \
      --output text 2>/dev/null)

    STATE=$(aws elbv2 describe-load-balancers \
      --region "$REGION" \
      --load-balancer-arns "$LB" \
      --query 'LoadBalancers[0].State.Code' \
      --output text 2>/dev/null)

    log "Processing LB: $NAME | Type: $TYPE | State: $STATE | ARN: $LB"

    delete_listeners_for_lb "$LB"

    log "Deleting load balancer: $NAME"
    run_cmd aws elbv2 delete-load-balancer \
      --region "$REGION" \
      --load-balancer-arn "$LB"
  done
}

delete_leftover_target_groups() {
  log "Checking leftover Kubernetes/EKS target groups"

  TGS=$(aws elbv2 describe-target-groups \
    --region "$REGION" \
    --query "TargetGroups[?starts_with(TargetGroupName, 'k8s-') || contains(TargetGroupName, 'backend') || contains(TargetGroupName, 'ingress') || contains(TargetGroupName, 'internal')].TargetGroupArn" \
    --output text 2>/dev/null)

  if [ -z "$TGS" ]; then
    log "No leftover Kubernetes/EKS-looking target groups found."
    return
  fi

  for TG in $TGS; do
    NAME=$(aws elbv2 describe-target-groups \
      --region "$REGION" \
      --target-group-arns "$TG" \
      --query 'TargetGroups[0].TargetGroupName' \
      --output text 2>/dev/null)

    log "Attempting target group delete: $NAME | ARN: $TG"

    run_cmd aws elbv2 delete-target-group \
      --region "$REGION" \
      --target-group-arn "$TG"
  done
}

summary_check() {
  log "======================================"
  log "ELBV2 SUMMARY CHECK"
  log "======================================"

  log "Load balancers:"
  aws elbv2 describe-load-balancers \
    --region "$REGION" \
    --query 'LoadBalancers[].{Name:LoadBalancerName,Type:Type,State:State.Code,DNS:DNSName}' \
    --output table 2>&1 | tee -a "$LOG_FILE"

  log "Target groups:"
  aws elbv2 describe-target-groups \
    --region "$REGION" \
    --query 'TargetGroups[].{Name:TargetGroupName,Port:Port,Protocol:Protocol,Type:TargetType}' \
    --output table 2>&1 | tee -a "$LOG_FILE"
}

main() {
  log "Starting leftover ELBV2 cleanup"
  log "Mode: $MODE"
  log "Region: $REGION"

  confirm_mode

  delete_leftover_load_balancers

  log "Waiting 60 seconds for load balancer deletion propagation"
  sleep 60

  delete_leftover_target_groups

  summary_check

  log "Leftover ELBV2 cleanup completed."
  log "Log file: $LOG_FILE"
}

#!/bin/bash

set -u

REGION="us-east-1"
MODE="${1:-dry-run}"
OUT_DIR="eks-deletion-logs"
RUN_DATE=$(date '+%Y-%m-%d')
LOG_FILE="$OUT_DIR/delete-eks-safe-resources-$RUN_DATE.log"

mkdir -p "$OUT_DIR"

log() {
  echo "[$RUN_DATE] $*" | tee -a "$LOG_FILE"
}

run_cmd() {
  log "COMMAND: $*"

  if [ "$MODE" = "--execute" ]; then
    "$@" 2>&1 | tee -a "$LOG_FILE" || true
  else
    log "DRY-RUN ONLY"
  fi

  echo | tee -a "$LOG_FILE"
}

confirm_mode() {
  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY-RUN mode."
    log "To execute: ./delete-eks-associated-safe-resources.sh --execute"
    echo
  else
    log "RUNNING IN EXECUTE MODE."
    read -p "Type DELETE-EKS to continue: " CONFIRM
    if [ "$CONFIRM" != "DELETE-EKS" ]; then
      log "Cancelled."
      exit 1
    fi
  fi
}

delete_load_balancers() {
  CLUSTER="$1"
  VPC_ID="$2"

  log "Deleting EKS/Kubernetes-associated load balancers for cluster: $CLUSTER"

  LBS=$(aws elbv2 describe-load-balancers \
    --region "$REGION" \
    --query 'LoadBalancers[].LoadBalancerArn' \
    --output text 2>/dev/null)

  for LB_ARN in $LBS; do
    LB_JSON=$(aws elbv2 describe-load-balancers \
      --region "$REGION" \
      --load-balancer-arns "$LB_ARN" \
      --query 'LoadBalancers[0]' \
      --output json 2>/dev/null)

    LB_NAME=$(echo "$LB_JSON" | jq -r '.LoadBalancerName')
    LB_VPC=$(echo "$LB_JSON" | jq -r '.VpcId')

    TAG_MATCH=$(aws elbv2 describe-tags \
      --region "$REGION" \
      --resource-arns "$LB_ARN" \
      --query "TagDescriptions[0].Tags[?Key=='kubernetes.io/cluster/$CLUSTER' || Key=='elbv2.k8s.aws/cluster' || Key=='ingress.k8s.aws/cluster']" \
      --output text 2>/dev/null)

    NAME_MATCH="false"
    if echo "$LB_NAME" | grep -qiE "^k8s-|^internal-k8s-|$CLUSTER"; then
      NAME_MATCH="true"
    fi

    if [ "$LB_VPC" = "$VPC_ID" ] && { [ -n "$TAG_MATCH" ] || [ "$NAME_MATCH" = "true" ]; }; then
      log "Deleting load balancer: $LB_NAME | $LB_ARN"
      run_cmd aws elbv2 delete-load-balancer \
        --region "$REGION" \
        --load-balancer-arn "$LB_ARN"
    fi
  done
}

delete_target_groups() {
  CLUSTER="$1"
  VPC_ID="$2"

  log "Deleting EKS/Kubernetes-associated target groups for cluster: $CLUSTER"

  TGS=$(aws elbv2 describe-target-groups \
    --region "$REGION" \
    --query 'TargetGroups[].TargetGroupArn' \
    --output text 2>/dev/null)

  for TG_ARN in $TGS; do
    TG_JSON=$(aws elbv2 describe-target-groups \
      --region "$REGION" \
      --target-group-arns "$TG_ARN" \
      --query 'TargetGroups[0]' \
      --output json 2>/dev/null)

    TG_NAME=$(echo "$TG_JSON" | jq -r '.TargetGroupName')
    TG_VPC=$(echo "$TG_JSON" | jq -r '.VpcId')

    TAG_MATCH=$(aws elbv2 describe-tags \
      --region "$REGION" \
      --resource-arns "$TG_ARN" \
      --query "TagDescriptions[0].Tags[?Key=='kubernetes.io/cluster/$CLUSTER' || Key=='elbv2.k8s.aws/cluster' || Key=='ingress.k8s.aws/cluster']" \
      --output text 2>/dev/null)

    NAME_MATCH="false"
    if echo "$TG_NAME" | grep -qiE "^k8s-|$CLUSTER"; then
      NAME_MATCH="true"
    fi

    if [ "$TG_VPC" = "$VPC_ID" ] && { [ -n "$TAG_MATCH" ] || [ "$NAME_MATCH" = "true" ]; }; then
      log "Deleting target group: $TG_NAME | $TG_ARN"
      run_cmd aws elbv2 delete-target-group \
        --region "$REGION" \
        --target-group-arn "$TG_ARN"
    fi
  done
}

delete_nodegroups() {
  CLUSTER="$1"

  log "Deleting nodegroups for cluster: $CLUSTER"

  NODEGROUPS=$(aws eks list-nodegroups \
    --region "$REGION" \
    --cluster-name "$CLUSTER" \
    --query 'nodegroups[]' \
    --output text 2>/dev/null)

  if [ -z "$NODEGROUPS" ]; then
    log "No nodegroups found for cluster: $CLUSTER"
    return
  fi

  for NG in $NODEGROUPS; do
    STATUS=$(aws eks describe-nodegroup \
      --region "$REGION" \
      --cluster-name "$CLUSTER" \
      --nodegroup-name "$NG" \
      --query 'nodegroup.status' \
      --output text 2>/dev/null)

    log "Nodegroup found: $NG | Status: $STATUS"

    if [ "$STATUS" = "DELETING" ]; then
      log "Skipping nodegroup already deleting: $NG"
    else
      run_cmd aws eks delete-nodegroup \
        --region "$REGION" \
        --cluster-name "$CLUSTER" \
        --nodegroup-name "$NG"
    fi
  done
}

wait_for_nodegroups() {
  CLUSTER="$1"

  if [ "$MODE" != "--execute" ]; then
    log "DRY-RUN ONLY: would wait for nodegroups to delete"
    return
  fi

  log "Waiting for nodegroups to be deleted for cluster: $CLUSTER"

  while true; do
    NODEGROUPS=$(aws eks list-nodegroups \
      --region "$REGION" \
      --cluster-name "$CLUSTER" \
      --query 'nodegroups[]' \
      --output text 2>/dev/null)

    if [ -z "$NODEGROUPS" ]; then
      log "All nodegroups deleted for cluster: $CLUSTER"
      break
    fi

    log "Still waiting. Remaining nodegroups: $NODEGROUPS"
    sleep 30
  done
}

delete_cluster() {
  CLUSTER="$1"

  log "Deleting EKS cluster: $CLUSTER"

  STATUS=$(aws eks describe-cluster \
    --region "$REGION" \
    --name "$CLUSTER" \
    --query 'cluster.status' \
    --output text 2>/dev/null)

  if [ "$STATUS" = "DELETING" ]; then
    log "Cluster already deleting: $CLUSTER"
  else
    run_cmd aws eks delete-cluster \
      --region "$REGION" \
      --name "$CLUSTER"
  fi
}

delete_cloudwatch_log_groups() {
  CLUSTER="$1"

  log "Deleting CloudWatch log groups for cluster: $CLUSTER"

  LOG_GROUPS=$(aws logs describe-log-groups \
    --region "$REGION" \
    --log-group-name-prefix "/aws/eks/$CLUSTER" \
    --query 'logGroups[].logGroupName' \
    --output text 2>/dev/null)

  if [ -z "$LOG_GROUPS" ]; then
    log "No EKS CloudWatch log groups found for cluster: $CLUSTER"
    return
  fi

  for LG in $LOG_GROUPS; do
    log "Deleting log group: $LG"
    run_cmd aws logs delete-log-group \
      --region "$REGION" \
      --log-group-name "$LG"
  done
}

main() {
  log "Starting safe EKS-associated resource deletion"
  log "Region: $REGION"
  log "Mode: $MODE"

  confirm_mode

  CLUSTERS=$(aws eks list-clusters \
    --region "$REGION" \
    --query 'clusters[]' \
    --output text 2>/dev/null)

  if [ -z "$CLUSTERS" ]; then
    log "No EKS clusters found."
    exit 0
  fi

  for CLUSTER in $CLUSTERS; do
    log "======================================"
    log "Processing cluster: $CLUSTER"
    log "======================================"

    VPC_ID=$(aws eks describe-cluster \
      --region "$REGION" \
      --name "$CLUSTER" \
      --query 'cluster.resourcesVpcConfig.vpcId' \
      --output text 2>/dev/null)

    log "Cluster VPC: $VPC_ID"

    delete_load_balancers "$CLUSTER" "$VPC_ID"

    log "Waiting briefly for load balancer deletion propagation"
    if [ "$MODE" = "--execute" ]; then
      sleep 60
    fi

    delete_target_groups "$CLUSTER" "$VPC_ID"

    delete_nodegroups "$CLUSTER"
    wait_for_nodegroups "$CLUSTER"

    delete_cluster "$CLUSTER"

    delete_cloudwatch_log_groups "$CLUSTER"
  done

  log "Safe EKS deletion script completed."
  log "Log file: $LOG_FILE"
}

#!/bin/bash

set -u

REGION="us-east-1"
OUT_DIR="eks-before-evidence"
RUN_DATE=$(date '+%Y-%m-%d')
REPORT="$OUT_DIR/eks-before-associated-resources-$RUN_DATE.txt"
CSV="$OUT_DIR/eks-before-associated-resources-$RUN_DATE.csv"

mkdir -p "$OUT_DIR"

echo "Category,Cluster,ResourceType,ResourceName,ResourceId,Region,VPC,AssociationReason,DeleteCandidate" > "$CSV"

log() {
  echo "$*" | tee -a "$REPORT"
}

csv() {
  echo "\"$1\",\"$2\",\"$3\",\"$4\",\"$5\",\"$6\",\"$7\",\"$8\",\"$9\"" >> "$CSV"
}

capture_window() {
  NAME="$1"

  echo
  echo "======================================"
  echo "Captured section: $NAME"
  echo "Date: $RUN_DATE"
  echo "Region: $REGION"
  echo "======================================"
  echo

  sleep 2
  screencapture -x -i -w "$OUT_DIR/${NAME}.png"
}

section_capture() {
  NAME="$1"
  TITLE="$2"

  clear
  echo "EKS BEFORE Evidence Capture"
  echo "Account: $(aws sts get-caller-identity --query Account --output text 2>/dev/null || echo UNKNOWN)"
  echo "Region: $REGION"
  echo "Check: $TITLE"
  echo "Date: $RUN_DATE"
  echo "======================================"
  echo
  cat "$REPORT"
  capture_window "$NAME"
}

log "EKS Associated Resource Discovery - BEFORE"
log "Date: $RUN_DATE"
log "Region: $REGION"
log "======================================"
log ""

CLUSTERS=$(aws eks list-clusters \
  --region "$REGION" \
  --query 'clusters[]' \
  --output text 2>/dev/null)

if [ -z "$CLUSTERS" ]; then
  log "No EKS clusters found."
  section_capture "01-eks-before-summary" "EKS Associated Resources"
  exit 0
fi

for CLUSTER in $CLUSTERS; do
  log ""
  log "======================================"
  log "Cluster: $CLUSTER"
  log "======================================"

  VPC_ID=$(aws eks describe-cluster \
    --region "$REGION" \
    --name "$CLUSTER" \
    --query 'cluster.resourcesVpcConfig.vpcId' \
    --output text 2>/dev/null)

  CLUSTER_SG=$(aws eks describe-cluster \
    --region "$REGION" \
    --name "$CLUSTER" \
    --query 'cluster.resourcesVpcConfig.clusterSecurityGroupId' \
    --output text 2>/dev/null)

  STATUS=$(aws eks describe-cluster \
    --region "$REGION" \
    --name "$CLUSTER" \
    --query 'cluster.status' \
    --output text 2>/dev/null)

  VERSION=$(aws eks describe-cluster \
    --region "$REGION" \
    --name "$CLUSTER" \
    --query 'cluster.version' \
    --output text 2>/dev/null)

  log "Status: $STATUS"
  log "Version: $VERSION"
  log "VPC: $VPC_ID"
  log "Cluster SG: $CLUSTER_SG"
  log ""

  csv "EKS" "$CLUSTER" "Cluster" "$CLUSTER" "$CLUSTER" "$REGION" "$VPC_ID" "EKS cluster object" "Yes"

  if [ "$CLUSTER_SG" != "None" ] && [ -n "$CLUSTER_SG" ]; then
    csv "Security" "$CLUSTER" "Cluster Security Group" "$CLUSTER_SG" "$CLUSTER_SG" "$REGION" "$VPC_ID" "clusterSecurityGroupId from EKS describe-cluster" "Later"
  fi

  log "Nodegroups:"
  NODEGROUPS=$(aws eks list-nodegroups \
    --region "$REGION" \
    --cluster-name "$CLUSTER" \
    --query 'nodegroups[]' \
    --output text 2>/dev/null)

  if [ -z "$NODEGROUPS" ]; then
    log "  None"
  else
    for NG in $NODEGROUPS; do
      NG_STATUS=$(aws eks describe-nodegroup \
        --region "$REGION" \
        --cluster-name "$CLUSTER" \
        --nodegroup-name "$NG" \
        --query 'nodegroup.status' \
        --output text 2>/dev/null)

      ASGS=$(aws eks describe-nodegroup \
        --region "$REGION" \
        --cluster-name "$CLUSTER" \
        --nodegroup-name "$NG" \
        --query 'nodegroup.resources.autoScalingGroups[].name' \
        --output text 2>/dev/null)

      log "  - $NG | Status: $NG_STATUS | ASG: ${ASGS:-None}"

      csv "EKS" "$CLUSTER" "Nodegroup" "$NG" "$NG" "$REGION" "$VPC_ID" "Listed from EKS nodegroup API" "Yes"

      for ASG in $ASGS; do
        csv "Compute" "$CLUSTER" "Auto Scaling Group" "$ASG" "$ASG" "$REGION" "$VPC_ID" "Returned by describe-nodegroup" "Yes"
      done
    done
  fi

  log ""
  log "Load Balancers associated with EKS/Kubernetes:"
  LBS=$(aws elbv2 describe-load-balancers \
    --region "$REGION" \
    --query 'LoadBalancers[].LoadBalancerArn' \
    --output text 2>/dev/null)

  LB_FOUND="false"

  for LB_ARN in $LBS; do
    LB_JSON=$(aws elbv2 describe-load-balancers \
      --region "$REGION" \
      --load-balancer-arns "$LB_ARN" \
      --query 'LoadBalancers[0]' \
      --output json 2>/dev/null)

    LB_NAME=$(echo "$LB_JSON" | jq -r '.LoadBalancerName')
    LB_VPC=$(echo "$LB_JSON" | jq -r '.VpcId')
    LB_TYPE=$(echo "$LB_JSON" | jq -r '.Type')
    LB_STATE=$(echo "$LB_JSON" | jq -r '.State.Code')

    TAG_MATCH=$(aws elbv2 describe-tags \
      --region "$REGION" \
      --resource-arns "$LB_ARN" \
      --query "TagDescriptions[0].Tags[?Key=='kubernetes.io/cluster/$CLUSTER' || Key=='elbv2.k8s.aws/cluster' || Key=='ingress.k8s.aws/cluster'].{Key:Key,Value:Value}" \
      --output text 2>/dev/null)

    NAME_MATCH="false"
    if echo "$LB_NAME" | grep -qiE "^k8s-|^internal-k8s-|$CLUSTER"; then
      NAME_MATCH="true"
    fi

    if [ "$LB_VPC" = "$VPC_ID" ] && { [ -n "$TAG_MATCH" ] || [ "$NAME_MATCH" = "true" ]; }; then
      LB_FOUND="true"
      log "  - $LB_NAME | Type: $LB_TYPE | State: $LB_STATE | VPC: $LB_VPC | ARN: $LB_ARN"
      csv "Networking" "$CLUSTER" "Load Balancer" "$LB_NAME" "$LB_ARN" "$REGION" "$LB_VPC" "Same VPC plus Kubernetes/EKS tag or name pattern" "Yes"
    fi
  done

  if [ "$LB_FOUND" = "false" ]; then
    log "  None found"
  fi

  log ""
  log "Target Groups associated with EKS/Kubernetes:"
  TGS=$(aws elbv2 describe-target-groups \
    --region "$REGION" \
    --query 'TargetGroups[].TargetGroupArn' \
    --output text 2>/dev/null)

  TG_FOUND="false"

  for TG_ARN in $TGS; do
    TG_JSON=$(aws elbv2 describe-target-groups \
      --region "$REGION" \
      --target-group-arns "$TG_ARN" \

set -u

RUN_DATE=$(date '+%Y-%m-%d')

AFTER_DIR="resource-inventory-after"
REPORT_DIR="before-after-report"

mkdir -p "$BEFORE_DIR" "$AFTER_DIR" "$REPORT_DIR"

MASTER="$REPORT_DIR/before-after-summary.csv"

echo "Service,Resource Type,Before Count,After Count,Status,Date" > "$MASTER"

count_lines() {
  FILE="$1"

  if [ ! -f "$FILE" ]; then
    echo 0
    return
  fi

  tail -n +2 "$FILE" | grep -v '^$' | wc -l | tr -d ' '
}

write_header() {
  FILE="$1"
  echo "Service,Resource Type,Resource Name,Status,Region,Additional Detail,Evidence Date" > "$FILE"
}

capture_before_after_csvs() {
  PHASE="$1"
  OUT_DIR="$2"

  RDS_INST="$OUT_DIR/01-rds-instances-$PHASE.csv"
  write_header "$RDS_INST"
  aws rds describe-db-instances \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .DBInstances[]? |
    ["RDS","DB Instance",.DBInstanceIdentifier,.DBInstanceStatus,$region,.Engine,$date] | @csv
  ' >> "$RDS_INST"

  RDS_CLUSTERS="$OUT_DIR/02-rds-clusters-$PHASE.csv"
  write_header "$RDS_CLUSTERS"
  aws rds describe-db-clusters \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .DBClusters[]? |
    ["RDS","DB Cluster",.DBClusterIdentifier,.Status,$region,.Engine,$date] | @csv
  ' >> "$RDS_CLUSTERS"

  RDS_SNAPS="$OUT_DIR/03-rds-manual-snapshots-$PHASE.csv"
  write_header "$RDS_SNAPS"
  aws rds describe-db-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .DBSnapshots[]? |
    ["RDS","Manual DB Snapshot",.DBSnapshotIdentifier,.Status,$region,.Engine,$date] | @csv
  ' >> "$RDS_SNAPS"

  RDS_CLUSTER_SNAPS="$OUT_DIR/04-rds-manual-cluster-snapshots-$PHASE.csv"
  write_header "$RDS_CLUSTER_SNAPS"
  aws rds describe-db-cluster-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .DBClusterSnapshots[]? |
    ["RDS","Manual Cluster Snapshot",.DBClusterSnapshotIdentifier,.Status,$region,.Engine,$date] | @csv
  ' >> "$RDS_CLUSTER_SNAPS"

  S3="$OUT_DIR/05-s3-buckets-$PHASE.csv"
  write_header "$S3"
  aws s3api list-buckets \
    --output json | jq -r --arg date "$RUN_DATE" '
    .Buckets[]? |
    ["S3","Bucket",.Name,"Present","Global",(.CreationDate // ""),$date] | @csv
  ' >> "$S3"

  DDB="$OUT_DIR/06-dynamodb-tables-$PHASE.csv"
  write_header "$DDB"
  aws dynamodb list-tables \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .TableNames[]? |
    ["DynamoDB","Table",.,"Present",$region,"",$date] | @csv
  ' >> "$DDB"

  DDB_BACKUPS="$OUT_DIR/07-dynamodb-backups-$PHASE.csv"
  write_header "$DDB_BACKUPS"
  aws dynamodb list-backups \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .BackupSummaries[]? |
    ["DynamoDB","Backup",.BackupName,.BackupStatus,$region,.TableName,$date] | @csv
  ' >> "$DDB_BACKUPS"

  EBS_SNAPS="$OUT_DIR/08-ebs-snapshots-$PHASE.csv"
  write_header "$EBS_SNAPS"
  aws ec2 describe-snapshots \
    --region "$REGION" \
    --owner-ids self \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .Snapshots[]? |
    ["EBS","Snapshot",.SnapshotId,.State,$region,(.VolumeId // ""),$date] | @csv
  ' >> "$EBS_SNAPS"

  EFS="$OUT_DIR/09-efs-filesystems-$PHASE.csv"
  write_header "$EFS"
  aws efs describe-file-systems \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .FileSystems[]? |
    ["EFS","File System",.FileSystemId,.LifeCycleState,$region,("SizeBytes=" + ((.SizeInBytes.Value // 0) | tostring)),$date] | @csv
  ' >> "$EFS"

  BACKUP="$OUT_DIR/10-backup-vaults-$PHASE.csv"
  write_header "$BACKUP"
  aws backup list-backup-vaults \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .BackupVaultList[]? |
    ["AWS Backup","Backup Vault",.BackupVaultName,"Present",$region,("RecoveryPoints=" + ((.NumberOfRecoveryPoints // 0) | tostring)),$date] | @csv
  ' >> "$BACKUP"
}

add_summary_row() {
  SERVICE="$1"
  RESOURCE_TYPE="$2"
  BEFORE_FILE="$3"
  AFTER_FILE="$4"

  BEFORE_COUNT=$(count_lines "$BEFORE_FILE")
  AFTER_COUNT=$(count_lines "$AFTER_FILE")

  if [ "$AFTER_COUNT" = "0" ]; then
    STATUS="Cleared"
  elif [ "$BEFORE_COUNT" = "$AFTER_COUNT" ]; then
    STATUS="Still Present"
  else
    STATUS="Partially Cleared"
  fi

  echo "$SERVICE,$RESOURCE_TYPE,$BEFORE_COUNT,$AFTER_COUNT,$STATUS,$RUN_DATE" >> "$MASTER"
}

#!/bin/bash

set -u

REGION="us-east-1"
MODE="${1:-dry-run}"
OUT_DIR="second-pass-deletion-logs"
RUN_DATE=$(date '+%Y-%m-%d')
LOG_FILE="$OUT_DIR/second-pass-${RUN_DATE}.log"

mkdir -p "$OUT_DIR"

log() {
  echo "[$RUN_DATE] $*" | tee -a "$LOG_FILE"
}

run_cmd() {
  log "COMMAND: $*"

  if [ "$MODE" = "--execute" ]; then
    "$@" 2>&1 | tee -a "$LOG_FILE" || true
  else
    log "DRY-RUN ONLY"
  fi

  echo | tee -a "$LOG_FILE"
}

confirm_mode() {
  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY-RUN mode."
    log "To execute: ./delete-data-resources-second-pass.sh --execute"
    echo
  else
    log "RUNNING SECOND PASS IN EXECUTE MODE."
    read -p "Type SECOND-PASS-DELETE to continue: " CONFIRM
    if [ "$CONFIRM" != "SECOND-PASS-DELETE" ]; then
      log "Cancelled."
      exit 1
    fi
  fi
}

delete_remaining_rds_instances() {
  log "Checking remaining RDS DB instances"

  DBS=$(aws rds describe-db-instances \
    --region "$REGION" \
    --query 'DBInstances[].DBInstanceIdentifier' \
    --output text 2>/dev/null)

  if [ -z "$DBS" ]; then
    log "No RDS DB instances found."
    return
  fi

  for DB in $DBS; do
    STATUS=$(aws rds describe-db-instances \
      --region "$REGION" \
      --db-instance-identifier "$DB" \
      --query 'DBInstances[0].DBInstanceStatus' \
      --output text 2>/dev/null)

    log "RDS DB instance found: $DB, status: $STATUS"

    if [[ "$STATUS" == "deleting" ]]; then
      log "Skipping $DB because it is already deleting."
      continue
    fi

    run_cmd aws rds modify-db-instance \
      --region "$REGION" \
      --db-instance-identifier "$DB" \
      --no-deletion-protection \
      --apply-immediately

    run_cmd aws rds delete-db-instance \
      --region "$REGION" \
      --db-instance-identifier "$DB" \
      --skip-final-snapshot \
      --delete-automated-backups
  done
}

delete_remaining_rds_snapshots() {
  log "Checking remaining manual RDS DB snapshots"

  SNAPS=$(aws rds describe-db-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --query 'DBSnapshots[].DBSnapshotIdentifier' \
    --output text 2>/dev/null)

  if [ -z "$SNAPS" ]; then
    log "No manual RDS DB snapshots found."
    return
  fi

  for SNAP in $SNAPS; do
    STATUS=$(aws rds describe-db-snapshots \
      --region "$REGION" \
      --db-snapshot-identifier "$SNAP" \
      --query 'DBSnapshots[0].Status' \
      --output text 2>/dev/null)

    log "RDS snapshot found: $SNAP, status: $STATUS"

    if [[ "$STATUS" == "deleting" ]]; then
      log "Skipping $SNAP because it is already deleting."
      continue
    fi

    run_cmd aws rds delete-db-snapshot \
      --region "$REGION" \
      --db-snapshot-identifier "$SNAP"
  done
}

delete_remaining_dynamodb_tables() {
  log "Checking remaining DynamoDB tables"

  TABLES=$(aws dynamodb list-tables \
    --region "$REGION" \
    --query 'TableNames[]' \
    --output text 2>/dev/null)

  if [ -z "$TABLES" ]; then
    log "No DynamoDB tables found."
    return
  fi

  for TABLE in $TABLES; do
    STATUS=$(aws dynamodb describe-table \
      --region "$REGION" \
      --table-name "$TABLE" \
      --query 'Table.TableStatus' \
      --output text 2>/dev/null)

    log "DynamoDB table found: $TABLE, status: $STATUS"

    if [[ "$STATUS" == "DELETING" ]]; then
      log "Skipping $TABLE because it is already deleting."
      continue
    fi

    run_cmd aws dynamodb delete-table \
      --region "$REGION" \
      --table-name "$TABLE"
  done
}

delete_remaining_ebs_snapshots() {
  log "Checking remaining EBS snapshots"

  SNAPS=$(aws ec2 describe-snapshots \
    --region "$REGION" \
    --owner-ids self \
    --query 'Snapshots[].SnapshotId' \
    --output text 2>/dev/null)

  if [ -z "$SNAPS" ]; then
    log "No EBS snapshots found."
    return
  fi

  for SNAP in $SNAPS; do
    STATE=$(aws ec2 describe-snapshots \
      --region "$REGION" \
      --snapshot-ids "$SNAP" \
      --query 'Snapshots[0].State' \
      --output text 2>/dev/null)

    log "EBS snapshot found: $SNAP, state: $STATE"

    if [[ "$STATE" == "pending" ]]; then
      log "Skipping $SNAP because snapshot is pending."
      continue
    fi

    run_cmd aws ec2 delete-snapshot \
      --region "$REGION" \
      --snapshot-id "$SNAP"
  done
}

delete_remaining_efs_filesystems() {
  log "Checking remaining EFS file systems"

  FILESYSTEMS=$(aws efs describe-file-systems \
    --region "$REGION" \
    --query 'FileSystems[].FileSystemId' \
    --output text 2>/dev/null)

  if [ -z "$FILESYSTEMS" ]; then
    log "No EFS file systems found."
    return
  fi

  for FS in $FILESYSTEMS; do
    STATE=$(aws efs describe-file-systems \
      --region "$REGION" \
      --file-system-id "$FS" \
      --query 'FileSystems[0].LifeCycleState' \
      --output text 2>/dev/null)

    log "EFS file system found: $FS, state: $STATE"

    MOUNTS=$(aws efs describe-mount-targets \
      --region "$REGION" \
      --file-system-id "$FS" \
      --query 'MountTargets[].MountTargetId' \
      --output text 2>/dev/null)

    for MT in $MOUNTS; do
      MT_STATE=$(aws efs describe-mount-targets \
        --region "$REGION" \
        --mount-target-id "$MT" \
        --query 'MountTargets[0].LifeCycleState' \
        --output text 2>/dev/null)

      log "EFS mount target found: $MT, state: $MT_STATE"

      if [[ "$MT_STATE" == "deleting" ]]; then
        log "Skipping mount target $MT because it is already deleting."
      else
        run_cmd aws efs delete-mount-target \
          --region "$REGION" \
          --mount-target-id "$MT"
      fi
    done

    REMAINING_MOUNTS=$(aws efs describe-mount-targets \
      --region "$REGION" \
      --file-system-id "$FS" \
      --query 'length(MountTargets)' \
      --output text 2>/dev/null)

    if [ "$REMAINING_MOUNTS" != "0" ]; then
      log "Skipping EFS file system $FS until mount targets are fully deleted."
      continue
    fi

    run_cmd aws efs delete-file-system \
      --region "$REGION" \
      --file-system-id "$FS"
  done
}

delete_remaining_backup_recovery_points_and_vaults() {
  log "Checking remaining AWS Backup vaults"

  VAULTS=$(aws backup list-backup-vaults \
    --region "$REGION" \
    --query 'BackupVaultList[].BackupVaultName' \
    --output text 2>/dev/null)

  if [ -z "$VAULTS" ]; then
    log "No AWS Backup vaults found."
    return
  fi

  for VAULT in $VAULTS; do
    log "Backup vault found: $VAULT"

    RECOVERY_POINTS=$(aws backup list-recovery-points-by-backup-vault \
      --region "$REGION" \
      --backup-vault-name "$VAULT" \
      --query 'RecoveryPoints[].RecoveryPointArn' \
      --output text 2>/dev/null)

    if [ -n "$RECOVERY_POINTS" ]; then
      for RP in $RECOVERY_POINTS; do
        log "Recovery point found: $RP"

        run_cmd aws backup delete-recovery-point \
          --region "$REGION" \
          --backup-vault-name "$VAULT" \
          --recovery-point-arn "$RP"
      done
    else
      log "No recovery points in vault: $VAULT"
    fi

    REMAINING_RPS=$(aws backup list-recovery-points-by-backup-vault \
      --region "$REGION" \
      --backup-vault-name "$VAULT" \
      --query 'length(RecoveryPoints)' \
      --output text 2>/dev/null)

    if [ "$REMAINING_RPS" != "0" ]; then
      log "Skipping backup vault $VAULT because recovery points still remain."
      continue
    fi

    run_cmd aws backup delete-backup-vault \
      --region "$REGION" \
      --backup-vault-name "$VAULT"
  done
}

#!/bin/bash

set -u

BUCKET="${1:-}"
MODE="${2:-dry-run}"
OUT_DIR="s3-single-bucket-deletion-logs"
RUN_DATE=$(date '+%Y-%m-%d')
LOG_FILE="$OUT_DIR/delete-${BUCKET}-${RUN_DATE}.log"
MAX_RETRIES=3
SLEEP_SECONDS=5

mkdir -p "$OUT_DIR"

log() {
  echo "[$RUN_DATE] $*" | tee -a "$LOG_FILE"
}

usage() {
  echo "Usage:"
  echo "./delete-single-s3-bucket.sh <bucket-name>            # dry run"
  echo "./delete-single-s3-bucket.sh <bucket-name> --execute  # actual deletion"
}

retry_cmd() {
  local attempt=1

  while [ "$attempt" -le "$MAX_RETRIES" ]; do
    log "Attempt $attempt/$MAX_RETRIES: $*"

    if "$@" 2>&1 | tee -a "$LOG_FILE"; then
      return 0
    fi

    log "Command failed. Retrying after ${SLEEP_SECONDS}s..."
    sleep "$SLEEP_SECONDS"
    attempt=$((attempt + 1))
  done

  log "FAILED after $MAX_RETRIES attempts: $*"
  return 1
}

confirm_mode() {
  if [ -z "$BUCKET" ]; then
    usage
    exit 1
  fi

  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY-RUN mode."
    log "No deletion will happen."
    log "To execute: ./delete-single-s3-bucket.sh $BUCKET --execute"
    echo
  else
    log "RUNNING IN EXECUTE MODE."
    log "Bucket targeted for deletion: $BUCKET"
    read -p "Type DELETE-S3 to continue: " CONFIRM

    if [ "$CONFIRM" != "DELETE-S3" ]; then
      log "Cancelled."
      exit 1
    fi
  fi
}

run_or_dry() {
  if [ "$MODE" = "--execute" ]; then
    retry_cmd "$@"
  else
    log "DRY-RUN ONLY: $*"
  fi
}

check_bucket() {
  log "Checking bucket access: $BUCKET"

  if ! aws s3api head-bucket --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE"; then
    log "Bucket not accessible or does not exist."
    exit 1
  fi
}

capture_bucket_metadata() {
  log "Capturing bucket metadata"

  log "Bucket location:"
  aws s3api get-bucket-location --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true

  log "Bucket versioning:"
  aws s3api get-bucket-versioning --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true

  log "Object lock configuration:"
  aws s3api get-object-lock-configuration --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true

  log "Bucket policy:"
  aws s3api get-bucket-policy --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true
}

abort_multipart_uploads() {
  log "Checking multipart uploads"

  aws s3api list-multipart-uploads \
    --bucket "$BUCKET" \
    --query 'Uploads[].{Key:Key,UploadId:UploadId}' \
    --output json > "$OUT_DIR/${BUCKET}-multipart.json" 2>/dev/null || echo "[]" > "$OUT_DIR/${BUCKET}-multipart.json"

  COUNT=$(jq 'length' "$OUT_DIR/${BUCKET}-multipart.json" 2>/dev/null || echo 0)

  log "Multipart uploads found: $COUNT"

  if [ "$COUNT" = "0" ]; then
    return
  fi

  jq -c '.[]' "$OUT_DIR/${BUCKET}-multipart.json" | while read -r upload; do
    KEY=$(echo "$upload" | jq -r '.Key')
    UPLOAD_ID=$(echo "$upload" | jq -r '.UploadId')

    run_or_dry aws s3api abort-multipart-upload \
      --bucket "$BUCKET" \
      --key "$KEY" \
      --upload-id "$UPLOAD_ID"
  done
}

delete_versions_batch() {
  log "Deleting object versions in batches"

  while true; do
    aws s3api list-object-versions \
      --bucket "$BUCKET" \
      --max-items 1000 \
      --output json \
      --query '{Objects: Versions[].{Key:Key,VersionId:VersionId}}' \
      > "$OUT_DIR/${BUCKET}-versions-batch.json" 2>/dev/null || echo '{"Objects":[]}' > "$OUT_DIR/${BUCKET}-versions-batch.json"

    COUNT=$(jq '.Objects | length' "$OUT_DIR/${BUCKET}-versions-batch.json" 2>/dev/null || echo 0)
    log "Object versions in current batch: $COUNT"

    if [ "$COUNT" = "0" ]; then
      break
    fi

    run_or_dry aws s3api delete-objects \
      --bucket "$BUCKET" \
      --delete "file://$OUT_DIR/${BUCKET}-versions-batch.json"

    if [ "$MODE" != "--execute" ]; then
      break
    fi

    sleep 2
  done
}

delete_markers_batch() {
  log "Deleting delete markers in batches"

  while true; do
    aws s3api list-object-versions \
      --bucket "$BUCKET" \
      --max-items 1000 \
      --output json \
      --query '{Objects: DeleteMarkers[].{Key:Key,VersionId:VersionId}}' \
      > "$OUT_DIR/${BUCKET}-delete-markers-batch.json" 2>/dev/null || echo '{"Objects":[]}' > "$OUT_DIR/${BUCKET}-delete-markers-batch.json"

    COUNT=$(jq '.Objects | length' "$OUT_DIR/${BUCKET}-delete-markers-batch.json" 2>/dev/null || echo 0)
    log "Delete markers in current batch: $COUNT"

    if [ "$COUNT" = "0" ]; then
      break
    fi

    run_or_dry aws s3api delete-objects \
      --bucket "$BUCKET" \
      --delete "file://$OUT_DIR/${BUCKET}-delete-markers-batch.json"

    if [ "$MODE" != "--execute" ]; then
      break
    fi

    sleep 2
  done
}

delete_non_versioned_objects() {
  log "Deleting remaining non-versioned/current objects"

  run_or_dry aws s3 rm "s3://$BUCKET" --recursive
}

delete_bucket() {
  log "Attempting final bucket delete"

  run_or_dry aws s3api delete-bucket --bucket "$BUCKET"
}

validate_bucket_deleted() {
  log "Validating bucket deletion"

  if aws s3api head-bucket --bucket "$BUCKET" 2>/dev/null; then
    log "VALIDATION FAILED: bucket still exists or is still accessible."
    log "Run diagnostics:"
    log "aws s3api list-object-versions --bucket $BUCKET --max-items 10"
    log "aws s3api get-object-lock-configuration --bucket $BUCKET"
    log "aws s3api get-bucket-policy --bucket $BUCKET"
    return 1
  else
    log "VALIDATION PASSED: bucket is deleted or no longer accessible."
    return 0
  fi
}

main() {
  confirm_mode

  log "Starting single bucket deletion"
  log "Bucket: $BUCKET"
  log "Mode: $MODE"

  check_bucket
  capture_bucket_metadata
  abort_multipart_uploads
  delete_versions_batch
  delete_markers_batch
  delete_non_versioned_objects
  delete_bucket

  if [ "$MODE" = "--execute" ]; then
    validate_bucket_deleted
  fi

  log "Completed single bucket deletion process"
  log "Log file: $LOG_FILE"
}

#!/bin/bash

set -u

REGION="us-east-1"
OUT_DIR="deletion-logs"
RUN_DATE=$(date '+%Y-%m-%d')
MODE="${1:-dry-run}"

mkdir -p "$OUT_DIR"

LOG_FILE="$OUT_DIR/delete-data-resources-${RUN_DATE}.log"

log() {
  echo "[$RUN_DATE] $*" | tee -a "$LOG_FILE"
}

run_cmd() {
  log "COMMAND: $*"

  if [ "$MODE" = "--execute" ]; then
    "$@" 2>&1 | tee -a "$LOG_FILE"
  else
    log "DRY-RUN ONLY: command not executed"
  fi

  echo | tee -a "$LOG_FILE"
}

confirm_mode() {
  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY-RUN mode."
    log "No resources will be deleted."
    log "To delete resources, run: ./delete-data-resources.sh --execute"
    echo
  else
    log "RUNNING IN EXECUTE MODE."
    log "Resources will be deleted from region: $REGION"
    echo
    read -p "Type DELETE to continue: " CONFIRM
    if [ "$CONFIRM" != "DELETE" ]; then
      log "Execution cancelled."
      exit 1
    fi
  fi
}

delete_rds_instances() {
  log "Deleting RDS DB instances"

  DBS=$(aws rds describe-db-instances \
    --region "$REGION" \
    --query 'DBInstances[].DBInstanceIdentifier' \
    --output text 2>/dev/null)

  for DB in $DBS; do
    log "Disabling deletion protection for RDS DB instance: $DB"

    run_cmd aws rds modify-db-instance \
      --region "$REGION" \
      --db-instance-identifier "$DB" \
      --no-deletion-protection \
      --apply-immediately

    log "Deleting RDS DB instance without final snapshot: $DB"

    run_cmd aws rds delete-db-instance \
      --region "$REGION" \
      --db-instance-identifier "$DB" \
      --skip-final-snapshot \
      --delete-automated-backups
  done
}

delete_rds_snapshots() {
  log "Deleting manual RDS DB snapshots"

  SNAPSHOTS=$(aws rds describe-db-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --query 'DBSnapshots[].DBSnapshotIdentifier' \
    --output text 2>/dev/null)

  for SNAP in $SNAPSHOTS; do
    log "Deleting RDS manual snapshot: $SNAP"

    run_cmd aws rds delete-db-snapshot \
      --region "$REGION" \
      --db-snapshot-identifier "$SNAP"
  done
}

delete_dynamodb_tables() {
  log "Deleting DynamoDB tables"

  TABLES=$(aws dynamodb list-tables \
    --region "$REGION" \
    --query 'TableNames[]' \
    --output text 2>/dev/null)

  for TABLE in $TABLES; do
    log "Deleting DynamoDB table: $TABLE"

    run_cmd aws dynamodb delete-table \
      --region "$REGION" \
      --table-name "$TABLE"
  done
}

delete_ebs_snapshots() {
  log "Deleting self-owned EBS snapshots"

  SNAPSHOTS=$(aws ec2 describe-snapshots \
    --region "$REGION" \
    --owner-ids self \
    --query 'Snapshots[].SnapshotId' \
    --output text 2>/dev/null)

  for SNAP in $SNAPSHOTS; do
    log "Deleting EBS snapshot: $SNAP"

    run_cmd aws ec2 delete-snapshot \
      --region "$REGION" \
      --snapshot-id "$SNAP"
  done
}

delete_efs_filesystems() {
  log "Deleting EFS file systems"

  FILESYSTEMS=$(aws efs describe-file-systems \
    --region "$REGION" \
    --query 'FileSystems[].FileSystemId' \
    --output text 2>/dev/null)

  for FS in $FILESYSTEMS; do
    log "Processing EFS file system: $FS"

    MOUNTS=$(aws efs describe-mount-targets \
      --region "$REGION" \
      --file-system-id "$FS" \
      --query 'MountTargets[].MountTargetId' \
      --output text 2>/dev/null)

    for MT in $MOUNTS; do
      log "Deleting EFS mount target: $MT"

      run_cmd aws efs delete-mount-target \
        --region "$REGION" \
        --mount-target-id "$MT"
    done

    log "Deleting EFS file system: $FS"

    run_cmd aws efs delete-file-system \
      --region "$REGION" \
      --file-system-id "$FS"
  done
}

delete_backup_recovery_points_and_vaults() {
  log "Deleting AWS Backup recovery points and vaults"

  VAULTS=$(aws backup list-backup-vaults \
    --region "$REGION" \
    --query 'BackupVaultList[].BackupVaultName' \
    --output text 2>/dev/null)

  for VAULT in $VAULTS; do
    log "Processing backup vault: $VAULT"

    RECOVERY_POINTS=$(aws backup list-recovery-points-by-backup-vault \
      --region "$REGION" \
      --backup-vault-name "$VAULT" \
      --query 'RecoveryPoints[].RecoveryPointArn' \
      --output text 2>/dev/null)

    for RP in $RECOVERY_POINTS; do
      log "Deleting recovery point: $RP"

      run_cmd aws backup delete-recovery-point \
        --region "$REGION" \
        --backup-vault-name "$VAULT" \
        --recovery-point-arn "$RP"
    done

    log "Deleting backup vault: $VAULT"

    run_cmd aws backup delete-backup-vault \
      --region "$REGION" \
      --backup-vault-name "$VAULT"
  done
}

delete_s3_buckets() {
  log "Deleting S3 buckets including versioned objects and delete markers"

  BUCKETS=$(aws s3api list-buckets \
    --query 'Buckets[].Name' \
    --output text 2>/dev/null)

  for BUCKET in $BUCKETS; do
    log "Processing S3 bucket: $BUCKET"

    if [ "$MODE" = "--execute" ]; then
      log "Deleting all object versions and delete markers from bucket: $BUCKET"

      aws s3api list-object-versions \
        --bucket "$BUCKET" \
        --output json \
        --query '{Objects: Versions[].{Key:Key,VersionId:VersionId}}' \
        > "$OUT_DIR/${BUCKET}-versions.json" 2>/dev/null || true

      if grep -q '"Key"' "$OUT_DIR/${BUCKET}-versions.json"; then
        aws s3api delete-objects \
          --bucket "$BUCKET" \
          --delete "file://$OUT_DIR/${BUCKET}-versions.json" \
          2>&1 | tee -a "$LOG_FILE" || true
      fi

      aws s3api list-object-versions \
        --bucket "$BUCKET" \
        --output json \
        --query '{Objects: DeleteMarkers[].{Key:Key,VersionId:VersionId}}' \
        > "$OUT_DIR/${BUCKET}-delete-markers.json" 2>/dev/null || true

      if grep -q '"Key"' "$OUT_DIR/${BUCKET}-delete-markers.json"; then
        aws s3api delete-objects \
          --bucket "$BUCKET" \
          --delete "file://$OUT_DIR/${BUCKET}-delete-markers.json" \
          2>&1 | tee -a "$LOG_FILE" || true
      fi

      log "Removing remaining non-versioned objects from bucket: $BUCKET"
      aws s3 rm "s3://$BUCKET" --recursive 2>&1 | tee -a "$LOG_FILE" || true

      log "Deleting bucket: $BUCKET"
      aws s3api delete-bucket --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true
    else
      log "DRY-RUN ONLY: would empty and delete bucket: $BUCKET"
    fi

    echo | tee -a "$LOG_FILE"
  done
}

main() {
  log "Starting data resource deletion"
  log "Region: $REGION"
  log "Mode: $MODE"
  echo

  confirm_mode

  delete_rds_instances
  delete_rds_snapshots
  delete_dynamodb_tables
  delete_ebs_snapshots
  delete_efs_filesystems
  delete_backup_recovery_points_and_vaults
  delete_s3_buckets

  log "Deletion script completed"
  log "Log file: $LOG_FILE"
}
