#!/bin/bash

set -u

REGION="us-east-1"
OUT_DIR="deletion-logs"
RUN_DATE=$(date '+%Y-%m-%d')
MODE="${1:-dry-run}"

mkdir -p "$OUT_DIR"

LOG_FILE="$OUT_DIR/delete-data-resources-${RUN_DATE}.log"

log() {
  echo "[$RUN_DATE] $*" | tee -a "$LOG_FILE"
}

run_cmd() {
  log "COMMAND: $*"

  if [ "$MODE" = "--execute" ]; then
    "$@" 2>&1 | tee -a "$LOG_FILE"
  else
    log "DRY-RUN ONLY: command not executed"
  fi

  echo | tee -a "$LOG_FILE"
}

confirm_mode() {
  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY-RUN mode."
    log "No resources will be deleted."
    log "To delete resources, run: ./delete-data-resources.sh --execute"
    echo
  else
    log "RUNNING IN EXECUTE MODE."
    log "Resources will be deleted from region: $REGION"
    echo
    read -p "Type DELETE to continue: " CONFIRM
    if [ "$CONFIRM" != "DELETE" ]; then
      log "Execution cancelled."
      exit 1
    fi
  fi
}

delete_rds_instances() {
  log "Deleting RDS DB instances"

  DBS=$(aws rds describe-db-instances \
    --region "$REGION" \
    --query 'DBInstances[].DBInstanceIdentifier' \
    --output text 2>/dev/null)

  for DB in $DBS; do
    log "Disabling deletion protection for RDS DB instance: $DB"

    run_cmd aws rds modify-db-instance \
      --region "$REGION" \
      --db-instance-identifier "$DB" \
      --no-deletion-protection \
      --apply-immediately

    log "Deleting RDS DB instance without final snapshot: $DB"

    run_cmd aws rds delete-db-instance \
      --region "$REGION" \
      --db-instance-identifier "$DB" \
      --skip-final-snapshot \
      --delete-automated-backups
  done
}

delete_rds_snapshots() {
  log "Deleting manual RDS DB snapshots"

  SNAPSHOTS=$(aws rds describe-db-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --query 'DBSnapshots[].DBSnapshotIdentifier' \
    --output text 2>/dev/null)

  for SNAP in $SNAPSHOTS; do
    log "Deleting RDS manual snapshot: $SNAP"

    run_cmd aws rds delete-db-snapshot \
      --region "$REGION" \
      --db-snapshot-identifier "$SNAP"
  done
}

delete_dynamodb_tables() {
  log "Deleting DynamoDB tables"

  TABLES=$(aws dynamodb list-tables \
    --region "$REGION" \
    --query 'TableNames[]' \
    --output text 2>/dev/null)

  for TABLE in $TABLES; do
    log "Deleting DynamoDB table: $TABLE"

    run_cmd aws dynamodb delete-table \
      --region "$REGION" \
      --table-name "$TABLE"
  done
}

delete_ebs_snapshots() {
  log "Deleting self-owned EBS snapshots"

  SNAPSHOTS=$(aws ec2 describe-snapshots \
    --region "$REGION" \
    --owner-ids self \
    --query 'Snapshots[].SnapshotId' \
    --output text 2>/dev/null)

  for SNAP in $SNAPSHOTS; do
    log "Deleting EBS snapshot: $SNAP"

    run_cmd aws ec2 delete-snapshot \
      --region "$REGION" \
      --snapshot-id "$SNAP"
  done
}

delete_efs_filesystems() {
  log "Deleting EFS file systems"

  FILESYSTEMS=$(aws efs describe-file-systems \
    --region "$REGION" \
    --query 'FileSystems[].FileSystemId' \
    --output text 2>/dev/null)

  for FS in $FILESYSTEMS; do
    log "Processing EFS file system: $FS"

    MOUNTS=$(aws efs describe-mount-targets \
      --region "$REGION" \
      --file-system-id "$FS" \
      --query 'MountTargets[].MountTargetId' \
      --output text 2>/dev/null)

    for MT in $MOUNTS; do
      log "Deleting EFS mount target: $MT"

      run_cmd aws efs delete-mount-target \
        --region "$REGION" \
        --mount-target-id "$MT"
    done

    log "Deleting EFS file system: $FS"

    run_cmd aws efs delete-file-system \
      --region "$REGION" \
      --file-system-id "$FS"
  done
}

delete_backup_recovery_points_and_vaults() {
  log "Deleting AWS Backup recovery points and vaults"

  VAULTS=$(aws backup list-backup-vaults \
    --region "$REGION" \
    --query 'BackupVaultList[].BackupVaultName' \
    --output text 2>/dev/null)

  for VAULT in $VAULTS; do
    log "Processing backup vault: $VAULT"

    RECOVERY_POINTS=$(aws backup list-recovery-points-by-backup-vault \
      --region "$REGION" \
      --backup-vault-name "$VAULT" \
      --query 'RecoveryPoints[].RecoveryPointArn' \
      --output text 2>/dev/null)

    for RP in $RECOVERY_POINTS; do
      log "Deleting recovery point: $RP"

      run_cmd aws backup delete-recovery-point \
        --region "$REGION" \
        --backup-vault-name "$VAULT" \
        --recovery-point-arn "$RP"
    done

    log "Deleting backup vault: $VAULT"

    run_cmd aws backup delete-backup-vault \
      --region "$REGION" \
      --backup-vault-name "$VAULT"
  done
}

delete_s3_buckets() {
  log "Deleting S3 buckets including versioned objects and delete markers"

  BUCKETS=$(aws s3api list-buckets \
    --query 'Buckets[].Name' \
    --output text 2>/dev/null)

  for BUCKET in $BUCKETS; do
    log "Processing S3 bucket: $BUCKET"

    if [ "$MODE" = "--execute" ]; then
      log "Deleting all object versions and delete markers from bucket: $BUCKET"

      aws s3api list-object-versions \
        --bucket "$BUCKET" \
        --output json \
        --query '{Objects: Versions[].{Key:Key,VersionId:VersionId}}' \
        > "$OUT_DIR/${BUCKET}-versions.json" 2>/dev/null || true

      if grep -q '"Key"' "$OUT_DIR/${BUCKET}-versions.json"; then
        aws s3api delete-objects \
          --bucket "$BUCKET" \
          --delete "file://$OUT_DIR/${BUCKET}-versions.json" \
          2>&1 | tee -a "$LOG_FILE" || true
      fi

      aws s3api list-object-versions \
        --bucket "$BUCKET" \
        --output json \
        --query '{Objects: DeleteMarkers[].{Key:Key,VersionId:VersionId}}' \
        > "$OUT_DIR/${BUCKET}-delete-markers.json" 2>/dev/null || true

      if grep -q '"Key"' "$OUT_DIR/${BUCKET}-delete-markers.json"; then
        aws s3api delete-objects \
          --bucket "$BUCKET" \
          --delete "file://$OUT_DIR/${BUCKET}-delete-markers.json" \
          2>&1 | tee -a "$LOG_FILE" || true
      fi

      log "Removing remaining non-versioned objects from bucket: $BUCKET"
      aws s3 rm "s3://$BUCKET" --recursive 2>&1 | tee -a "$LOG_FILE" || true

      log "Deleting bucket: $BUCKET"
      aws s3api delete-bucket --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true
    else
      log "DRY-RUN ONLY: would empty and delete bucket: $BUCKET"
    fi

    echo | tee -a "$LOG_FILE"
  done
}

main() {
  log "Starting data resource deletion"
  log "Region: $REGION"
  log "Mode: $MODE"
  echo

  confirm_mode

  delete_rds_instances
  delete_rds_snapshots
  delete_dynamodb_tables
  delete_ebs_snapshots
  delete_efs_filesystems
  delete_backup_recovery_points_and_vaults
  delete_s3_buckets

  log "Deletion script completed"
  log "Log file: $LOG_FILE"
}

main

#!/bin/bash

set -u

REGION="us-east-1"
OUT_DIR="resource-inventory"
RUN_DATE=$(date '+%Y-%m-%d')

mkdir -p "$OUT_DIR"

echo "Generating AWS resource inventory CSVs"
echo "Date: $RUN_DATE"
echo "Region: $REGION"

write_header() {
  FILE="$1"
  echo "Service,Resource Type,Resource Name,Status,Region,Additional Detail,Evidence Date" > "$FILE"
}

MASTER="$OUT_DIR/aws-resource-inventory-master.csv"
write_header "$MASTER"

append_csv() {
  FILE="$1"
  if [ -s "$FILE" ]; then
    tail -n +2 "$FILE" >> "$MASTER"
  fi
}

RDS_INST="$OUT_DIR/01-rds-instances.csv"
write_header "$RDS_INST"
aws rds describe-db-instances \
  --region "$REGION" \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .DBInstances[]? |
  ["RDS","DB Instance",.DBInstanceIdentifier,.DBInstanceStatus,$region,.Engine,$date] | @csv
' >> "$RDS_INST"
append_csv "$RDS_INST"

RDS_SNAP="$OUT_DIR/02-rds-manual-snapshots.csv"
write_header "$RDS_SNAP"
aws rds describe-db-snapshots \
  --region "$REGION" \
  --snapshot-type manual \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .DBSnapshots[]? |
  ["RDS","Manual Snapshot",.DBSnapshotIdentifier,.Status,$region,.Engine,$date] | @csv
' >> "$RDS_SNAP"
append_csv "$RDS_SNAP"

S3="$OUT_DIR/03-s3-buckets.csv"
write_header "$S3"
aws s3api list-buckets \
  --output json | jq -r --arg date "$RUN_DATE" '
  .Buckets[]? |
  ["S3","Bucket",.Name,"Present","Global",(.CreationDate // ""),$date] | @csv
' >> "$S3"
append_csv "$S3"

DDB="$OUT_DIR/04-dynamodb-tables.csv"
write_header "$DDB"
aws dynamodb list-tables \
  --region "$REGION" \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .TableNames[]? |
  ["DynamoDB","Table",.,"Present",$region,"",$date] | @csv
' >> "$DDB"
append_csv "$DDB"

DDB_BKP="$OUT_DIR/05-dynamodb-backups.csv"
write_header "$DDB_BKP"
aws dynamodb list-backups \
  --region "$REGION" \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .BackupSummaries[]? |
  ["DynamoDB","Backup",.BackupName,.BackupStatus,$region,.TableName,$date] | @csv
' >> "$DDB_BKP"
append_csv "$DDB_BKP"

EBS_SNAP="$OUT_DIR/06-ebs-snapshots.csv"
write_header "$EBS_SNAP"
aws ec2 describe-snapshots \
  --region "$REGION" \
  --owner-ids self \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .Snapshots[]? |
  ["EBS","Snapshot",.SnapshotId,.State,$region,(.VolumeId // ""),$date] | @csv
' >> "$EBS_SNAP"
append_csv "$EBS_SNAP"

EFS="$OUT_DIR/07-efs-filesystems.csv"
write_header "$EFS"
aws efs describe-file-systems \
  --region "$REGION" \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .FileSystems[]? |
  ["EFS","File System",.FileSystemId,.LifeCycleState,$region,("SizeBytes=" + ((.SizeInBytes.Value // 0) | tostring)),$date] | @csv
' >> "$EFS"
append_csv "$EFS"

BACKUP="$OUT_DIR/08-backup-vaults.csv"
write_header "$BACKUP"
aws backup list-backup-vaults \
  --region "$REGION" \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .BackupVaultList[]? |
  ["AWS Backup","Backup Vault",.BackupVaultName,"Present",$region,("RecoveryPoints=" + ((.NumberOfRecoveryPoints // 0) | tostring)),$date] | @csv
' >> "$BACKUP"
append_csv "$BACKUP"

EC2="$OUT_DIR/09-ec2-instances.csv"
write_header "$EC2"
aws ec2 describe-instances \
  --region "$REGION" \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .Reservations[]?.Instances[]? |
  ["EC2","Instance",.InstanceId,.State.Name,$region,.InstanceType,$date] | @csv
' >> "$EC2"
append_csv "$EC2"

EBS_VOL="$OUT_DIR/10-ebs-volumes.csv"
write_header "$EBS_VOL"
aws ec2 describe-volumes \
  --region "$REGION" \
  --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
  .Volumes[]? |
  ["EBS","Volume",.VolumeId,.State,$region,("SizeGiB=" + (.Size | tostring)),$date] | @csv
' >> "$EBS_VOL"
append_csv "$EBS_VOL"

echo
echo "CSV inventory generated:"
echo "$OUT_DIR"
echo
echo "Main Excel-openable file:"
echo "$MASTER"

#!/bin/bash

set -u

REGION="us-east-1"
OUT_DIR="evidence-before"
RUN_DATE=$(date '+%Y-%m-%d')

mkdir -p "$OUT_DIR"

capture_window() {
  NAME="$1"

  echo
  echo "======================================"
  echo "Captured section: $NAME"
  echo "Date: $RUN_DATE"
  echo "Region: $REGION"
  echo "======================================"
  echo

  sleep 2
  screencapture -x -i -w "$OUT_DIR/${NAME}.png"
}

run_and_capture() {
  NAME="$1"
  TITLE="$2"
  shift 2

  CMD="./$*"

  clear
  echo "AWS BEFORE Evidence Capture"
  echo "Account: $(aws sts get-caller-identity --query Account --output text 2>/dev/null || echo UNKNOWN)"
  echo "Region: $REGION"
  echo "Check: $TITLE"
  echo "Date: $RUN_DATE"
  echo "======================================"
  echo
  echo "Command:"
  echo "$CMD"
  echo "--------------------------------------"
  echo

  "$@" 2>&1 | tee "$OUT_DIR/${NAME}.txt"

  capture_window "$NAME"
}

run_and_capture \
  "01-rds-instances-before" \
  "RDS Instances" \
  aws rds describe-db-instances \
    --region "$REGION" \
    --query 'DBInstances[].{DB:DBInstanceIdentifier,Status:DBInstanceStatus,Engine:Engine}' \
    --output table

run_and_capture \
  "02-rds-snapshots-before" \
  "RDS Manual Snapshots" \
  aws rds describe-db-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --query 'DBSnapshots[].{Snapshot:DBSnapshotIdentifier,Status:Status,Engine:Engine}' \
    --output table

run_and_capture \
  "03-s3-buckets-before" \
  "S3 Buckets" \
  aws s3api list-buckets \
    --query 'Buckets[].{Bucket:Name,Created:CreationDate}' \
    --output table

run_and_capture \
  "04-dynamodb-tables-before" \
  "DynamoDB Tables" \
  aws dynamodb list-tables \
    --region "$REGION" \
    --query 'TableNames[]' \
    --output table

run_and_capture \
  "05-ebs-snapshots-before" \
  "EBS Snapshots" \
  aws ec2 describe-snapshots \
    --region "$REGION" \
    --owner-ids self \
    --query 'Snapshots[].{Snapshot:SnapshotId,State:State,Volume:VolumeId,Started:StartTime}' \
    --output table

run_and_capture \
  "06-efs-filesystems-before" \
  "EFS File Systems" \
  aws efs describe-file-systems \
    --region "$REGION" \
    --query 'FileSystems[].{FileSystem:FileSystemId,State:LifeCycleState,Created:CreationTime}' \
    --output table

run_and_capture \
  "07-backup-vaults-before" \
  "AWS Backup Vaults" \
  aws backup list-backup-vaults \
    --region "$REGION" \
    --query 'BackupVaultList[].{Vault:BackupVaultName,RecoveryPoints:NumberOfRecoveryPoints}' \
    --output table

echo
echo "BEFORE evidence capture completed"
echo "Saved in: $OUT_DIR"
