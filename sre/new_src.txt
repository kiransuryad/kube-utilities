#!/bin/bash

set -u

REGION="us-east-1"
OUT_DIR="eks-before-evidence"
RUN_DATE=$(date '+%Y-%m-%d')

mkdir -p "$OUT_DIR"

capture_window() {
  NAME="$1"

  echo
  echo "======================================"
  echo "Captured section: $NAME"
  echo "Date: $RUN_DATE"
  echo "Region: $REGION"
  echo "======================================"
  echo

  sleep 2
  screencapture -x -i -w "$OUT_DIR/${NAME}.png"
}

run_and_capture() {
  NAME="$1"
  TITLE="$2"
  shift 2

  CMD="./$*"

  clear

  echo "EKS BEFORE Evidence Capture"
  echo "Account: $(aws sts get-caller-identity --query Account --output text 2>/dev/null || echo UNKNOWN)"
  echo "Region: $REGION"
  echo "Check: $TITLE"
  echo "Date: $RUN_DATE"
  echo "======================================"
  echo
  echo "Command:"
  echo "$CMD"
  echo "--------------------------------------"
  echo

  "$@" 2>&1 | tee "$OUT_DIR/${NAME}.txt"

  capture_window "$NAME"
}

# ---------------------------------------------------
# EKS Clusters
# ---------------------------------------------------

run_and_capture \
  "01-eks-clusters-before" \
  "EKS Clusters" \
  aws eks list-clusters \
    --region "$REGION" \
    --output table

# ---------------------------------------------------
# EKS Cluster Details
# ---------------------------------------------------

CLUSTERS=$(aws eks list-clusters \
  --region "$REGION" \
  --query 'clusters[]' \
  --output text 2>/dev/null)

INDEX=2

for CLUSTER in $CLUSTERS; do

  PADDED=$(printf "%02d" $INDEX)

  run_and_capture \
    "${PADDED}-eks-cluster-${CLUSTER}-before" \
    "EKS Cluster Detail - ${CLUSTER}" \
    aws eks describe-cluster \
      --region "$REGION" \
      --name "$CLUSTER" \
      --query 'cluster.{Name:name,Status:status,Version:version,Endpoint:endpoint}' \
      --output table

  INDEX=$((INDEX + 1))

  PADDED=$(printf "%02d" $INDEX)

  run_and_capture \
    "${PADDED}-eks-nodegroups-${CLUSTER}-before" \
    "EKS Nodegroups - ${CLUSTER}" \
    aws eks list-nodegroups \
      --region "$REGION" \
      --cluster-name "$CLUSTER" \
      --output table

  INDEX=$((INDEX + 1))
done

# ---------------------------------------------------
# Auto Scaling Groups
# ---------------------------------------------------

run_and_capture \
  "10-autoscaling-groups-before" \
  "Auto Scaling Groups" \
  aws autoscaling describe-auto-scaling-groups \
    --region "$REGION" \
    --query 'AutoScalingGroups[].{Name:AutoScalingGroupName,Desired:DesiredCapacity,Min:MinSize,Max:MaxSize,Instances:length(Instances)}' \
    --output table

# ---------------------------------------------------
# EC2 Instances
# ---------------------------------------------------

run_and_capture \
  "11-ec2-instances-before" \
  "EC2 Instances" \
  aws ec2 describe-instances \
    --region "$REGION" \
    --query 'Reservations[].Instances[].{Instance:InstanceId,State:State.Name,Type:InstanceType,PrivateIP:PrivateIpAddress}' \
    --output table

# ---------------------------------------------------
# Load Balancers
# ---------------------------------------------------

run_and_capture \
  "12-load-balancers-before" \
  "Load Balancers" \
  aws elbv2 describe-load-balancers \
    --region "$REGION" \
    --query 'LoadBalancers[].{Name:LoadBalancerName,Type:Type,State:State.Code,DNS:DNSName}' \
    --output table

# ---------------------------------------------------
# Target Groups
# ---------------------------------------------------

run_and_capture \
  "13-target-groups-before" \
  "Target Groups" \
  aws elbv2 describe-target-groups \
    --region "$REGION" \
    --query 'TargetGroups[].{Name:TargetGroupName,Port:Port,Protocol:Protocol,Type:TargetType}' \
    --output table

# ---------------------------------------------------
# EBS Volumes
# ---------------------------------------------------

run_and_capture \
  "14-ebs-volumes-before" \
  "EBS Volumes" \
  aws ec2 describe-volumes \
    --region "$REGION" \
    --query 'Volumes[].{Volume:VolumeId,State:State,Size:Size,Type:VolumeType}' \
    --output table

# ---------------------------------------------------
# EBS Snapshots
# ---------------------------------------------------

run_and_capture \
  "15-ebs-snapshots-before" \
  "EBS Snapshots" \
  aws ec2 describe-snapshots \
    --region "$REGION" \
    --owner-ids self \
    --query 'Snapshots[].{Snapshot:SnapshotId,State:State,Volume:VolumeId}' \
    --output table

# ---------------------------------------------------
# EFS
# ---------------------------------------------------

run_and_capture \
  "16-efs-filesystems-before" \
  "EFS File Systems" \
  aws efs describe-file-systems \
    --region "$REGION" \
    --query 'FileSystems[].{FileSystem:FileSystemId,State:LifeCycleState,Encrypted:Encrypted}' \
    --output table

# ---------------------------------------------------
# Security Groups
# ---------------------------------------------------

run_and_capture \
  "17-security-groups-before" \
  "Security Groups" \
  aws ec2 describe-security-groups \
    --region "$REGION" \
    --query 'SecurityGroups[].{Group:GroupName,Id:GroupId,VPC:VpcId}' \
    --output table

# ---------------------------------------------------
# ENIs
# ---------------------------------------------------

run_and_capture \
  "18-network-interfaces-before" \
  "Network Interfaces" \
  aws ec2 describe-network-interfaces \
    --region "$REGION" \
    --query 'NetworkInterfaces[].{ENI:NetworkInterfaceId,Status:Status,PrivateIP:PrivateIpAddress,Type:InterfaceType}' \
    --output table

# ---------------------------------------------------
# CloudWatch Log Groups
# ---------------------------------------------------

run_and_capture \
  "19-cloudwatch-log-groups-before" \
  "CloudWatch Log Groups" \
  aws logs describe-log-groups \
    --region "$REGION" \
    --query 'logGroups[].{LogGroup:logGroupName,Retention:retentionInDays}' \
    --output table

echo
echo "EKS BEFORE evidence capture completed"
echo "Saved in: $OUT_DIR"

#!/bin/bash

set -u

REGION="us-east-1"
RUN_DATE=$(date '+%Y-%m-%d')

BEFORE_DIR="resource-inventory-before"
AFTER_DIR="resource-inventory-after"
REPORT_DIR="before-after-report"

mkdir -p "$BEFORE_DIR" "$AFTER_DIR" "$REPORT_DIR"

MASTER="$REPORT_DIR/before-after-summary.csv"

echo "Service,Resource Type,Before Count,After Count,Status,Date" > "$MASTER"

count_lines() {
  FILE="$1"

  if [ ! -f "$FILE" ]; then
    echo 0
    return
  fi

  tail -n +2 "$FILE" | grep -v '^$' | wc -l | tr -d ' '
}

write_header() {
  FILE="$1"
  echo "Service,Resource Type,Resource Name,Status,Region,Additional Detail,Evidence Date" > "$FILE"
}

capture_before_after_csvs() {
  PHASE="$1"
  OUT_DIR="$2"

  RDS_INST="$OUT_DIR/01-rds-instances-$PHASE.csv"
  write_header "$RDS_INST"
  aws rds describe-db-instances \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .DBInstances[]? |
    ["RDS","DB Instance",.DBInstanceIdentifier,.DBInstanceStatus,$region,.Engine,$date] | @csv
  ' >> "$RDS_INST"

  RDS_CLUSTERS="$OUT_DIR/02-rds-clusters-$PHASE.csv"
  write_header "$RDS_CLUSTERS"
  aws rds describe-db-clusters \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .DBClusters[]? |
    ["RDS","DB Cluster",.DBClusterIdentifier,.Status,$region,.Engine,$date] | @csv
  ' >> "$RDS_CLUSTERS"

  RDS_SNAPS="$OUT_DIR/03-rds-manual-snapshots-$PHASE.csv"
  write_header "$RDS_SNAPS"
  aws rds describe-db-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .DBSnapshots[]? |
    ["RDS","Manual DB Snapshot",.DBSnapshotIdentifier,.Status,$region,.Engine,$date] | @csv
  ' >> "$RDS_SNAPS"

  RDS_CLUSTER_SNAPS="$OUT_DIR/04-rds-manual-cluster-snapshots-$PHASE.csv"
  write_header "$RDS_CLUSTER_SNAPS"
  aws rds describe-db-cluster-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .DBClusterSnapshots[]? |
    ["RDS","Manual Cluster Snapshot",.DBClusterSnapshotIdentifier,.Status,$region,.Engine,$date] | @csv
  ' >> "$RDS_CLUSTER_SNAPS"

  S3="$OUT_DIR/05-s3-buckets-$PHASE.csv"
  write_header "$S3"
  aws s3api list-buckets \
    --output json | jq -r --arg date "$RUN_DATE" '
    .Buckets[]? |
    ["S3","Bucket",.Name,"Present","Global",(.CreationDate // ""),$date] | @csv
  ' >> "$S3"

  DDB="$OUT_DIR/06-dynamodb-tables-$PHASE.csv"
  write_header "$DDB"
  aws dynamodb list-tables \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .TableNames[]? |
    ["DynamoDB","Table",.,"Present",$region,"",$date] | @csv
  ' >> "$DDB"

  DDB_BACKUPS="$OUT_DIR/07-dynamodb-backups-$PHASE.csv"
  write_header "$DDB_BACKUPS"
  aws dynamodb list-backups \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .BackupSummaries[]? |
    ["DynamoDB","Backup",.BackupName,.BackupStatus,$region,.TableName,$date] | @csv
  ' >> "$DDB_BACKUPS"

  EBS_SNAPS="$OUT_DIR/08-ebs-snapshots-$PHASE.csv"
  write_header "$EBS_SNAPS"
  aws ec2 describe-snapshots \
    --region "$REGION" \
    --owner-ids self \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .Snapshots[]? |
    ["EBS","Snapshot",.SnapshotId,.State,$region,(.VolumeId // ""),$date] | @csv
  ' >> "$EBS_SNAPS"

  EFS="$OUT_DIR/09-efs-filesystems-$PHASE.csv"
  write_header "$EFS"
  aws efs describe-file-systems \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .FileSystems[]? |
    ["EFS","File System",.FileSystemId,.LifeCycleState,$region,("SizeBytes=" + ((.SizeInBytes.Value // 0) | tostring)),$date] | @csv
  ' >> "$EFS"

  BACKUP="$OUT_DIR/10-backup-vaults-$PHASE.csv"
  write_header "$BACKUP"
  aws backup list-backup-vaults \
    --region "$REGION" \
    --output json | jq -r --arg region "$REGION" --arg date "$RUN_DATE" '
    .BackupVaultList[]? |
    ["AWS Backup","Backup Vault",.BackupVaultName,"Present",$region,("RecoveryPoints=" + ((.NumberOfRecoveryPoints // 0) | tostring)),$date] | @csv
  ' >> "$BACKUP"
}

add_summary_row() {
  SERVICE="$1"
  RESOURCE_TYPE="$2"
  BEFORE_FILE="$3"
  AFTER_FILE="$4"

  BEFORE_COUNT=$(count_lines "$BEFORE_FILE")
  AFTER_COUNT=$(count_lines "$AFTER_FILE")

  if [ "$AFTER_COUNT" = "0" ]; then
    STATUS="Cleared"
  elif [ "$BEFORE_COUNT" = "$AFTER_COUNT" ]; then
    STATUS="Still Present"
  else
    STATUS="Partially Cleared"
  fi

  echo "$SERVICE,$RESOURCE_TYPE,$BEFORE_COUNT,$AFTER_COUNT,$STATUS,$RUN_DATE" >> "$MASTER"
}

echo "Generating BEFORE inventory snapshot..."
capture_before_after_csvs "before" "$BEFORE_DIR"

echo "Generating AFTER inventory snapshot..."
capture_before_after_csvs "after" "$AFTER_DIR"

add_summary_row "RDS" "DB Instance" "$BEFORE_DIR/01-rds-instances-before.csv" "$AFTER_DIR/01-rds-instances-after.csv"
add_summary_row "RDS" "DB Cluster" "$BEFORE_DIR/02-rds-clusters-before.csv" "$AFTER_DIR/02-rds-clusters-after.csv"
add_summary_row "RDS" "Manual DB Snapshot" "$BEFORE_DIR/03-rds-manual-snapshots-before.csv" "$AFTER_DIR/03-rds-manual-snapshots-after.csv"
add_summary_row "RDS" "Manual Cluster Snapshot" "$BEFORE_DIR/04-rds-manual-cluster-snapshots-before.csv" "$AFTER_DIR/04-rds-manual-cluster-snapshots-after.csv"
add_summary_row "S3" "Bucket" "$BEFORE_DIR/05-s3-buckets-before.csv" "$AFTER_DIR/05-s3-buckets-after.csv"
add_summary_row "DynamoDB" "Table" "$BEFORE_DIR/06-dynamodb-tables-before.csv" "$AFTER_DIR/06-dynamodb-tables-after.csv"
add_summary_row "DynamoDB" "Backup" "$BEFORE_DIR/07-dynamodb-backups-before.csv" "$AFTER_DIR/07-dynamodb-backups-after.csv"
add_summary_row "EBS" "Snapshot" "$BEFORE_DIR/08-ebs-snapshots-before.csv" "$AFTER_DIR/08-ebs-snapshots-after.csv"
add_summary_row "EFS" "File System" "$BEFORE_DIR/09-efs-filesystems-before.csv" "$AFTER_DIR/09-efs-filesystems-after.csv"
add_summary_row "AWS Backup" "Backup Vault" "$BEFORE_DIR/10-backup-vaults-before.csv" "$AFTER_DIR/10-backup-vaults-after.csv"

cat "$BEFORE_DIR"/*.csv > "$REPORT_DIR/before-resource-inventory-combined.csv"
cat "$AFTER_DIR"/*.csv > "$REPORT_DIR/after-resource-inventory-combined.csv"

echo
echo "Before/after report generated:"
echo "$REPORT_DIR"
echo
echo "Main summary:"
echo "$MASTER"

#!/bin/bash

set -u

REGION="us-east-1"
MODE="${1:-dry-run}"
OUT_DIR="second-pass-deletion-logs"
RUN_DATE=$(date '+%Y-%m-%d')
LOG_FILE="$OUT_DIR/second-pass-${RUN_DATE}.log"

mkdir -p "$OUT_DIR"

log() {
  echo "[$RUN_DATE] $*" | tee -a "$LOG_FILE"
}

run_cmd() {
  log "COMMAND: $*"

  if [ "$MODE" = "--execute" ]; then
    "$@" 2>&1 | tee -a "$LOG_FILE" || true
  else
    log "DRY-RUN ONLY"
  fi

  echo | tee -a "$LOG_FILE"
}

confirm_mode() {
  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY-RUN mode."
    log "To execute: ./delete-data-resources-second-pass.sh --execute"
    echo
  else
    log "RUNNING SECOND PASS IN EXECUTE MODE."
    read -p "Type SECOND-PASS-DELETE to continue: " CONFIRM
    if [ "$CONFIRM" != "SECOND-PASS-DELETE" ]; then
      log "Cancelled."
      exit 1

  for FS in $FILESYSTEMS; do
    STATE=$(aws efs describe-file-systems \
      --region "$REGION" \
      --file-system-id "$FS" \
      --query 'FileSystems[0].LifeCycleState' \
      --output text 2>/dev/null)

    log "EFS file system found: $FS, state: $STATE"

    MOUNTS=$(aws efs describe-mount-targets \
      --region "$REGION" \
      --file-system-id "$FS" \
      --query 'MountTargets[].MountTargetId' \
      --output text 2>/dev/null)

    for MT in $MOUNTS; do
      MT_STATE=$(aws efs describe-mount-targets \
        --region "$REGION" \
        --mount-target-id "$MT" \
        --query 'MountTargets[0].LifeCycleState' \
        --output text 2>/dev/null)

      log "EFS mount target found: $MT, state: $MT_STATE"

      if [[ "$MT_STATE" == "deleting" ]]; then
        log "Skipping mount target $MT because it is already deleting."
      else
        run_cmd aws efs delete-mount-target \
          --region "$REGION" \
          --mount-target-id "$MT"
      fi
    done

    REMAINING_MOUNTS=$(aws efs describe-mount-targets \
      --region "$REGION" \
      --file-system-id "$FS" \
      --query 'length(MountTargets)' \
      --output text 2>/dev/null)

      --region "$REGION" \

#!/bin/bash

set -u

BUCKET="${1:-}"
MODE="${2:-dry-run}"
OUT_DIR="s3-single-bucket-deletion-logs"
RUN_DATE=$(date '+%Y-%m-%d')
LOG_FILE="$OUT_DIR/delete-${BUCKET}-${RUN_DATE}.log"
MAX_RETRIES=3
SLEEP_SECONDS=5

mkdir -p "$OUT_DIR"

log() {
  echo "[$RUN_DATE] $*" | tee -a "$LOG_FILE"
}

usage() {
  echo "Usage:"
  echo "./delete-single-s3-bucket.sh <bucket-name>            # dry run"
  echo "./delete-single-s3-bucket.sh <bucket-name> --execute  # actual deletion"
}

retry_cmd() {
  local attempt=1

  while [ "$attempt" -le "$MAX_RETRIES" ]; do
    log "Attempt $attempt/$MAX_RETRIES: $*"

    if "$@" 2>&1 | tee -a "$LOG_FILE"; then
      return 0
    fi

    log "Command failed. Retrying after ${SLEEP_SECONDS}s..."
    sleep "$SLEEP_SECONDS"
    attempt=$((attempt + 1))
  done

  log "FAILED after $MAX_RETRIES attempts: $*"
  return 1
}

confirm_mode() {
  if [ -z "$BUCKET" ]; then
    usage
    exit 1
  fi

  if [ "$MODE" != "--execute" ]; then
    log "Running in DRY-RUN mode."
    log "No deletion will happen."
    log "To execute: ./delete-single-s3-bucket.sh $BUCKET --execute"
    echo
  else
    log "RUNNING IN EXECUTE MODE."
    log "Bucket targeted for deletion: $BUCKET"
    read -p "Type DELETE-S3 to continue: " CONFIRM

    if [ "$CONFIRM" != "DELETE-S3" ]; then
      log "Cancelled."
      exit 1
    fi
  fi
}

run_or_dry() {
  if [ "$MODE" = "--execute" ]; then
    retry_cmd "$@"
  else
    log "DRY-RUN ONLY: $*"
  fi
}

check_bucket() {
  log "Checking bucket access: $BUCKET"

  if ! aws s3api head-bucket --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE"; then
    log "Bucket not accessible or does not exist."
    exit 1
  fi
}

capture_bucket_metadata() {
  log "Capturing bucket metadata"

  log "Bucket location:"
  aws s3api get-bucket-location --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true

  log "Bucket versioning:"
  aws s3api get-bucket-versioning --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true

  log "Object lock configuration:"
  aws s3api get-object-lock-configuration --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true

  log "Bucket policy:"
  aws s3api get-bucket-policy --bucket "$BUCKET" 2>&1 | tee -a "$LOG_FILE" || true
}

abort_multipart_uploads() {
  log "Checking multipart uploads"

  aws s3api list-multipart-uploads \
    --bucket "$BUCKET" \
    --query 'Uploads[].{Key:Key,UploadId:UploadId}' \
    --output json > "$OUT_DIR/${BUCKET}-multipart.json" 2>/dev/null || echo "[]" > "$OUT_DIR/${BUCKET}-multipart.json"

  COUNT=$(jq 'length' "$OUT_DIR/${BUCKET}-multipart.json" 2>/dev/null || echo 0)

  log "Multipart uploads found: $COUNT"

  if [ "$COUNT" = "0" ]; then
    return
  fi

  jq -c '.[]' "$OUT_DIR/${BUCKET}-multipart.json" | while read -r upload; do
    KEY=$(echo "$upload" | jq -r '.Key')
    UPLOAD_ID=$(echo "$upload" | jq -r '.UploadId')

    run_or_dry aws s3api abort-multipart-upload \
      --bucket "$BUCKET" \
      --key "$KEY" \
      --upload-id "$UPLOAD_ID"
  done
}

delete_versions_batch() {
  log "Deleting object versions in batches"

  while true; do
    aws s3api list-object-versions \
      --bucket "$BUCKET" \
      --max-items 1000 \
      --output json \
      --query '{Objects: Versions[].{Key:Key,VersionId:VersionId}}' \
      > "$OUT_DIR/${BUCKET}-versions-batch.json" 2>/dev/null || echo '{"Objects":[]}' > "$OUT_DIR/${BUCKET}-versions-batch.json"

    COUNT=$(jq '.Objects | length' "$OUT_DIR/${BUCKET}-versions-batch.json" 2>/dev/null || echo 0)
    log "Object versions in current batch: $COUNT"

    if [ "$COUNT" = "0" ]; then
      break
    fi

    run_or_dry aws s3api delete-objects \
      --bucket "$BUCKET" \
      --delete "file://$OUT_DIR/${BUCKET}-versions-batch.json"

    if [ "$MODE" != "--execute" ]; then
      break
    fi

    sleep 2
  done
}

delete_markers_batch() {
  log "Deleting delete markers in batches"

  while true; do
    aws s3api list-object-versions \
      --bucket "$BUCKET" \
      --max-items 1000 \
      --output json \
      --query '{Objects: DeleteMarkers[].{Key:Key,VersionId:VersionId}}' \
      > "$OUT_DIR/${BUCKET}-delete-markers-batch.json" 2>/dev/null || echo '{"Objects":[]}' > "$OUT_DIR/${BUCKET}-delete-markers-batch.json"

    COUNT=$(jq '.Objects | length' "$OUT_DIR/${BUCKET}-delete-markers-batch.json" 2>/dev/null || echo 0)
    log "Delete markers in current batch: $COUNT"

    if [ "$COUNT" = "0" ]; then
      break
    fi

    run_or_dry aws s3api delete-objects \
      --bucket "$BUCKET" \
      --delete "file://$OUT_DIR/${BUCKET}-delete-markers-batch.json"

    if [ "$MODE" != "--execute" ]; then
      break
    fi

    sleep 2
  done
}

delete_non_versioned_objects() {
  log "Deleting remaining non-versioned/current objects"

  run_or_dry aws s3 rm "s3://$BUCKET" --recursive
}

delete_bucket() {
  log "Attempting final bucket delete"

  run_or_dry aws s3api delete-bucket --bucket "$BUCKET"
}

validate_bucket_deleted() {
  log "Validating bucket deletion"

  if aws s3api head-bucket --bucket "$BUCKET" 2>/dev/null; then
    log "VALIDATION FAILED: bucket still exists or is still accessible."
    log "Run diagnostics:"
    log "aws s3api list-object-versions --bucket $BUCKET --max-items 10"
    log "aws s3api get-object-lock-configuration --bucket $BUCKET"
    log "aws s3api get-bucket-policy --bucket $BUCKET"
    return 1
  else
    log "VALIDATION PASSED: bucket is deleted or no longer accessible."
    return 0
  fi
}

main() {
  confirm_mode

  log "Starting single bucket deletion"
  log "Bucket: $BUCKET"
  log "Mode: $MODE"

  check_bucket
  capture_bucket_metadata
  abort_multipart_uploads
  delete_versions_batch
  delete_markers_batch
  delete_non_versioned_objects
  delete_bucket

  if [ "$MODE" = "--execute" ]; then
    validate_bucket_deleted
  fi

  log "Completed single bucket deletion process"
  log "Log file: $LOG_FILE"
}

#!/bin/bash

set -u

REGION="us-east-1"
OUT_DIR="evidence-before"
RUN_DATE=$(date '+%Y-%m-%d')

mkdir -p "$OUT_DIR"

capture_window() {
  NAME="$1"

  echo
  echo "======================================"
  echo "Captured section: $NAME"
  echo "Date: $RUN_DATE"
  echo "Region: $REGION"
  echo "======================================"
  echo

  sleep 2
  screencapture -x -i -w "$OUT_DIR/${NAME}.png"
}

run_and_capture() {
  NAME="$1"
  TITLE="$2"
  shift 2

  CMD="./$*"

  clear
  echo "AWS BEFORE Evidence Capture"
  echo "Account: $(aws sts get-caller-identity --query Account --output text 2>/dev/null || echo UNKNOWN)"
  echo "Region: $REGION"
  echo "Check: $TITLE"
  echo "Date: $RUN_DATE"
  echo "======================================"
  echo
  echo "Command:"
  echo "$CMD"
  echo "--------------------------------------"
  echo

  "$@" 2>&1 | tee "$OUT_DIR/${NAME}.txt"

  capture_window "$NAME"
}

run_and_capture \
  "01-rds-instances-before" \
  "RDS Instances" \
  aws rds describe-db-instances \
    --region "$REGION" \
    --query 'DBInstances[].{DB:DBInstanceIdentifier,Status:DBInstanceStatus,Engine:Engine}' \
    --output table

run_and_capture \
  "02-rds-snapshots-before" \
  "RDS Manual Snapshots" \
  aws rds describe-db-snapshots \
    --region "$REGION" \
    --snapshot-type manual \
    --query 'DBSnapshots[].{Snapshot:DBSnapshotIdentifier,Status:Status,Engine:Engine}' \
    --output table

run_and_capture \
  "03-s3-buckets-before" \
  "S3 Buckets" \
  aws s3api list-buckets \
    --query 'Buckets[].{Bucket:Name,Created:CreationDate}' \
    --output table

run_and_capture \
  "04-dynamodb-tables-before" \
  "DynamoDB Tables" \
  aws dynamodb list-tables \
    --region "$REGION" \
    --query 'TableNames[]' \
    --output table

run_and_capture \
  "05-ebs-snapshots-before" \
  "EBS Snapshots" \
  aws ec2 describe-snapshots \
    --region "$REGION" \
    --owner-ids self \
    --query 'Snapshots[].{Snapshot:SnapshotId,State:State,Volume:VolumeId,Started:StartTime}' \
    --output table

run_and_capture \
  "06-efs-filesystems-before" \
  "EFS File Systems" \
  aws efs describe-file-systems \
    --region "$REGION" \
    --query 'FileSystems[].{FileSystem:FileSystemId,State:LifeCycleState,Created:CreationTime}' \
    --output table

run_and_capture \
  "07-backup-vaults-before" \
  "AWS Backup Vaults" \
  aws backup list-backup-vaults \
    --region "$REGION" \
    --query 'BackupVaultList[].{Vault:BackupVaultName,RecoveryPoints:NumberOfRecoveryPoints}' \
    --output table

echo
echo "BEFORE evidence capture completed"
echo "Saved in: $OUT_DIR"
